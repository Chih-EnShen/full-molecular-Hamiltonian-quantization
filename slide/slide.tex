\section{Motivation}

\begin{frame}{Why is the molecular Hamiltonian non-trivial?}
\setstage{0}
The lab Hamiltonian couples \emph{all} particles through the Coulomb potential:
\[
H=\sum_j\frac{\mathbf P_j^2}{2M_j}+\sum_i\frac{\mathbf p_i^2}{2m_e}+V .
\]
To do molecular physics we want to separate
\[
\underbrace{\text{translation}}_{\text{COM}}\;\big|\;\underbrace{\text{rotation}}_{\text{Euler angles }\Omega}\;\big|\;\underbrace{\text{vibration}}_{\text{normal coords }Q}\;\big|\;\underbrace{\text{electrons}}_{\bar{\mathbf r}_i}.
\]
\pause
\begin{alertblock}{The catch}
Rotation + vibration are \emph{curvilinear} coordinates. Naively quantizing
$p\to-i\hbar\,\partial$ is \textbf{not} Hermitian there. Getting it right forces the
Podolsky quantization.
\end{alertblock}
\end{frame}

\begin{frame}{What we will \emph{not} approximate}
\setstage{0}
We keep everything exact until the very end:
\begin{itemize}
  \item \textbf{No} Born--Oppenheimer / clamped nuclei.
  \item \textbf{No} neglect of \emph{mass polarization}.
  \item \textbf{No} harmonic assumption.
\end{itemize}
\pause
The \emph{only} ingredients are:
\begin{enumerate}
  \item non-relativistic dynamics;
  \item the Eckart frame (a \emph{convention}, not an approximation);
  \item parametrizing the nuclear shape by $3N_{\mathrm{nucl}}-6$ normal coordinates (non-linear molecules).
\end{enumerate}
\vspace{0.5ex}
\centering\itshape The endpoint is the exact quantum-mechanical internal molecular Hamiltonian $\hat H_{\mathrm{int}}$ — the rigorous starting point
\end{frame}

\setstage{1}
\section{Classical Cartesian Hamiltonian}

\begin{frame}{The Lagrangian in the lab frame}
\setstage{1}
$N_{\mathrm{nucl}}$ nuclei ($M_j,Z_je,\mathbf R_j$) and $N_{\mathrm{elec}}$ electrons ($m_e,-e,\mathbf r_i$), all in the space-fixed frame; SI units. The Lagrangian is kinetic minus potential:
\begin{equation*}
\mathcal L = \sum_j\tfrac12 M_j\dot{\mathbf R}_j^{\,2} + \sum_i\tfrac12 m_e\dot{\mathbf r}_i^{\,2} - V(\mathbf R_j,\mathbf r_i).
\end{equation*}
Differentiate $\mathcal L$ with respect to each velocity to get the \alert{canonical momenta}:
\begin{equation*}
\boxed{\;\mathbf{P}_j \equiv \frac{\partial \mathcal L}{\partial\dot{\mathbf R}_j} = M_j\dot{\mathbf R}_j,
\qquad
\mathbf{p}_i \equiv \frac{\partial \mathcal L}{\partial\dot{\mathbf r}_i} = m_e\dot{\mathbf r}_i.\;}
\end{equation*}
Canonical brackets $\{R_{j\alpha},P_{k\beta}\}=\delta_{jk}\delta_{\alpha\beta}$, $\{r_{i\alpha},p_{j\beta}\}=\delta_{ij}\delta_{\alpha\beta}$.
\begin{itemize}
  \item Kinetic energy is diagonal and quadratic in Cartesian velocities: \emph{canonical momentum $=$ mass $\times$ velocity}.
\end{itemize}
\end{frame}

\begin{frame}{Legendre transform to the Hamiltonian}
\setstage{1}
The three-move recipe (we reuse it, less trivially, in a moment):
\begin{enumerate}
  \item \textbf{Define} $p_q\equiv\partial \mathcal L/\partial\dot q$ for every coordinate.
  \item \textbf{Invert} for every velocity as a function of the momenta.
  \item \textbf{Assemble} $H\equiv\sum_qp_q\dot q-\mathcal L$ and substitute.
\end{enumerate}
Here step 2 is immediate, $\dot{\mathbf R}_j=\mathbf P_j/M_j$, $\dot{\mathbf r}_i=\mathbf p_i/m_e$; step 3 gives
\begin{equation*}
H = \underbrace{\sum_{j}\frac{\mathbf{P}_j^{\,2}}{2M_j}}_{T_N} + \underbrace{\sum_{i}\frac{\mathbf{p}_i^{\,2}}{2m_e}}_{T_e} + V, \tag{1.1}
\end{equation*}
\begin{equation*}
V = \frac{1}{4\pi\epsilon_0}\left(\frac12\sum_{i\ne j}\frac{Z_i Z_j e^2}{R_{ij}} - \sum_{i,j}\frac{Z_j e^2}{|\mathbf{R}_j-\mathbf{r}_i|} + \frac12\sum_{i\ne j}\frac{e^2}{r_{ij}}\right). \tag{1.2}
\end{equation*}
\vspace{0.5ex}
\centering\footnotesize Everything that follows is an exact rewriting of (1.1)--(1.2).
\end{frame}

\setstage{1}
\section{Mass-centered coordinates}

\begin{frame}{Two centers of mass, two roles}
\setstage{1}
The translational coordinate and the internal reference point \alert{need not be the same}. Define
\begin{equation*}
M_N=\sum_jM_j,\qquad M_{\mathrm{tot}}=M_N+N_{\mathrm{elec}}m_e,
\end{equation*}
\begin{align*}
\mathbf{R}_{\mathrm{ncm}} &= \frac{1}{M_N}\sum_j M_j\mathbf{R}_j,
&\mathbf{R}_{\mathrm{cm}} &= \frac{1}{M_{\mathrm{tot}}}\Bigl(\sum_j M_j\mathbf{R}_j+m_e\sum_i\mathbf{r}_i\Bigr), \tag{A.2, A.3}
\end{align*}
and refer \emph{every} internal coordinate --- nuclear and electronic --- to the \emph{nuclear} CoM:
\begin{equation*}
\mathbf{R}'_j = \mathbf{R}_j-\mathbf{R}_{\mathrm{ncm}},
\qquad\mathbf{r}'_i = \mathbf{r}_i-\mathbf{R}_{\mathrm{ncm}},\qquad \textstyle\sum_j M_j\mathbf{R}'_j=0. \tag{A.4}
\end{equation*}
\begin{block}{The mixed choice}
\textbf{Translation $=$ total CoM} $\Rightarrow$ correct total mass, and (we will see) the COM--internal cross terms cancel identically.\\
\textbf{Internals referred to the nuclear CoM} $\Rightarrow$ Born--Oppenheimer-friendly coordinates, $V$ form-invariant, and the mass polarization appears \emph{by itself}.
\end{block}
\end{frame}

\begin{frame}{Inverting: every particle rides a shared drift}
\setstage{1}
Solve (A.3) for the nuclear CoM, then invert (A.4) for the lab positions:
\begin{equation*}
\mathbf R_{\mathrm{ncm}} = \mathbf R_{\mathrm{cm}} - \frac{m_e}{M_{\mathrm{tot}}}\sum_l\mathbf r'_l , \tag{A.5}
\end{equation*}
\begin{equation*}
\mathbf R_j = \mathbf R_{\mathrm{cm}}+\mathbf R'_j-\frac{m_e}{M_{\mathrm{tot}}}\sum_l\mathbf r'_l,
\qquad
\mathbf r_i = \mathbf R_{\mathrm{cm}}+\mathbf r'_i-\frac{m_e}{M_{\mathrm{tot}}}\sum_l\mathbf r'_l . \tag{A.6}
\end{equation*}
Both families ride on the \emph{same} drift --- the nuclear-CoM velocity:
\begin{equation*}
\boxed{\;\mathbf D \equiv \dot{\mathbf R}_{\mathrm{ncm}} = \dot{\mathbf R}_{\mathrm{cm}}-\frac{m_e}{M_{\mathrm{tot}}}\sum_l\dot{\mathbf r}'_l,
\qquad \dot{\mathbf R}_j=\mathbf D+\dot{\mathbf R}'_j,\qquad \dot{\mathbf r}_i=\mathbf D+\dot{\mathbf r}'_i.\;} \tag{A.7}
\end{equation*}
\centering\itshape Every lab velocity $=$ internal velocity $+$ one shared drift $\mathbf D$ --- just like passengers on a train.
\end{frame}

\begin{frame}{Kinetic energy in velocity form: the mass polarization appears}
\setstage{1}
Substitute (A.7) into $T=\sum_j\tfrac12M_j\dot{\mathbf R}_j^2+\sum_i\tfrac12m_e\dot{\mathbf r}_i^2$. The nuclear cross term dies by $\sum_jM_j\dot{\mathbf R}'_j=0$; the electronic one does \emph{not}:
\begin{equation*}
T_N=\tfrac12M_N\mathbf D^2+\tfrac12\sum_jM_j\dot{\mathbf R}_j^{'2}, \qquad
T_e=\tfrac12N_{\mathrm{elec}}m_e\mathbf D^2+m_e\mathbf D\!\cdot\!\sum_i\dot{\mathbf r}'_i+\tfrac12m_e\sum_i\dot{\mathbf r}_i^{'2}. \tag{A.8, A.9}
\end{equation*}
\pause
Expand $\mathbf D$ in the two $\mathbf D$-dependent pieces (A.11a,b): the $\dot{\mathbf R}_{\mathrm{cm}}\!\cdot\!\sum\dot{\mathbf r}'$ cross terms cancel \alert{identically}, and the quadratic pieces combine to one \emph{negative} remainder:
\begin{equation*}
\boxed{\;T = \tfrac12 M_{\mathrm{tot}}\dot{\mathbf R}_{\mathrm{cm}}^{2}
+ \tfrac12\sum_j M_j\dot{\mathbf R}_j^{'2}
+ \tfrac12 m_e\sum_i\dot{\mathbf r}_i^{'2}
- \frac{m_e^2}{2M_{\mathrm{tot}}}\Bigl(\sum_i\dot{\mathbf r}'_i\Bigr)^{2}.\;} \tag{A.12}
\end{equation*}
\end{frame}

\begin{frame}{Canonical momenta: $\partial \mathcal L/\partial\dot q$, and the tricky one}
\setstage{1}
With $\mathcal L=T-V$ ($V$ velocity-independent), differentiate the boxed $T$:
\begin{equation*}
\mathbf P_{\mathrm{cm}} = \frac{\partial \mathcal L}{\partial\dot{\mathbf R}_{\mathrm{cm}}} = M_{\mathrm{tot}}\dot{\mathbf R}_{\mathrm{cm}},
\qquad
\mathbf P'_j = \frac{\partial \mathcal L}{\partial\dot{\mathbf R}'_j} = M_j\dot{\mathbf R}'_j . \tag{A.13, A.17}
\end{equation*}
The electrons take care: \emph{two} terms of $T$ involve $\dot{\mathbf r}'_i$ (its own kinetic term, \emph{and} the mass-polarization sum), so
\begin{equation*}
\boxed{\;\mathbf p'_i \equiv \frac{\partial \mathcal L}{\partial\dot{\mathbf r}'_i}
= m_e\dot{\mathbf r}'_i - \frac{m_e^2}{M_{\mathrm{tot}}}\sum_l\dot{\mathbf r}'_l
\;\ne\; m_e\dot{\mathbf r}'_i .\;} \tag{A.14}
\end{equation*}
% \alert{Canonical $\ne$ mechanical:} moving \emph{one} electron drags the whole molecule's CoM, and $\mathbf r'_i$ is measured from the (recoiling) nuclear CoM --- every electron's momentum carries a shared correction.
\end{frame}

\begin{frame}{Inverting the electron relation}
\setstage{1}
Sum (A.14) over all electrons; with $M_{\mathrm{tot}}-N_{\mathrm{elec}}m_e=M_N$,
\begin{equation*}
\sum_i\mathbf p'_i = m_e\Bigl(1-\frac{N_{\mathrm{elec}}m_e}{M_{\mathrm{tot}}}\Bigr)\sum_l\dot{\mathbf r}'_l = \frac{m_eM_N}{M_{\mathrm{tot}}}\sum_l\dot{\mathbf r}'_l . \tag{A.15}
\end{equation*}
Solve for $\dot{\mathbf r}'_i$:
\begin{equation*}
\boxed{\;\dot{\mathbf r}'_i = \frac{\mathbf p'_i}{m_e} + \frac{1}{M_N}\sum_l\mathbf p'_l .\;} \tag{A.16}
\end{equation*}
% \centering\itshape Each electron's velocity $=$ its own momentum$/m_e$ $+$ one shared recoil --- the momentum-space mirror of the shared drift $\mathbf D$.
\end{frame}

\begin{frame}{Legendre transform: kinetic energy in momentum form}
\setstage{1}
CoM and nuclear terms are immediate: $\tfrac12M_{\mathrm{tot}}\dot{\mathbf R}_{\mathrm{cm}}^2=\mathbf P_{\mathrm{cm}}^2/2M_{\mathrm{tot}}$, $\tfrac12M_j\dot{\mathbf R}_j^{'2}=\mathbf P_j^{'2}/2M_j$. Substituting (A.16) into the electron sector (A.19, A.20) and collecting the $\bigl(\sum_l\mathbf p'_l\bigr)^2$ coefficients,
\begin{equation*}
\frac{1}{M_N}+\frac{N_{\mathrm{elec}}m_e-M_{\mathrm{tot}}}{2M_N^2} = \frac{1}{M_N}-\frac{M_N}{2M_N^2} = +\frac{1}{2M_N} . \tag{A.21}
\end{equation*}
As a result, the overall kinetic energy in momentum form is
% \alert{The negative velocity-form correction has become the positive mass polarization:}
\begin{equation*}
\boxed{\;T = \frac{\mathbf P_{\mathrm{cm}}^{2}}{2M_{\mathrm{tot}}} + \sum_j\frac{\mathbf P_j^{'2}}{2M_j} + \sum_i\frac{\mathbf p_i^{'2}}{2m_e} + \frac{1}{2M_N}\Bigl(\sum_i\mathbf p'_i\Bigr)^{2}.\;} \tag{A.22}
\end{equation*}
\end{frame}

\begin{frame}{The Hamiltonian --- and why the sign flips}
\setstage{1}
Both internal families share the reference point $\mathbf R_{\mathrm{ncm}}$, so $V(\mathbf R'_j,\mathbf r'_i)$ is unchanged in form. Add it to (A.22):
\begin{equation*}
\boxed{\;H = \frac{\mathbf{P}_{\mathrm{cm}}^{\,2}}{2M_{\mathrm{tot}}}
+\underbrace{\sum_j\frac{\mathbf{P}_j^{'2}}{2M_j}+\sum_i\frac{\mathbf{p}_i^{'2}}{2m_e}
+\frac{1}{2M_N}\Bigl(\sum_i\mathbf{p}'_i\Bigr)^{\!2}+V(\mathbf R'_j,\mathbf r'_i)}_{\textstyle H_{\mathrm{int}}\ (2.13)}.\;} \tag{2.12 $\equiv$ A.23}
\end{equation*}
\footnotesize Exact for any $\mathbf P_{\mathrm{cm}}$: no rest frame, no boost, no cross-coupling --- agrees with the paper's generating-function route (Sec.\ 2).\normalsize
\centering
\end{frame}

\setstage{2}
\section{Body-fixed frame and the Lagrangian}

\begin{frame}{Body-fixed frame: rotate with the nuclei}
\setstage{2}
Introduce a rotation $\mathbf S(\Omega)$ (Euler angles $\Omega=\phi,\theta,\chi$) and body-fixed (BF) coordinates:
\begin{equation*}
\mathbf{R}'_j=\mathbf{S}(\Omega)\,\bar{\mathbf{R}}_j(Q),\qquad \mathbf{r}'_i=\mathbf{S}(\Omega)\,\bar{\mathbf{r}}_i. \tag{3.1}
\end{equation*}
The \emph{Eckart} conditions fix $\mathbf S$ from the \emph{nuclei}; the nuclear shape is rectilinear mass-weighted \alert{normal coordinates} $Q_b$:
\begin{equation*}
\bar{\mathbf{R}}_j(Q)=\bar{\mathbf{R}}_j^{\mathrm{eq}}+\frac{1}{\sqrt{M_j}}\sum_b\mathbf{l}_{jb}\,Q_b ,
\qquad \sum_j\mathbf{l}_{ja}\!\cdot\!\mathbf{l}_{jb}=\delta_{ab}. \tag{3.1a}
\end{equation*}
The BF angular velocity $\boldsymbol{\omega}$ is defined by
\begin{equation*}
(\mathbf{S}^{T}\dot{\mathbf{S}}) \mathbf v = \boldsymbol{\omega} \times \mathbf v
\end{equation*}
\begin{itemize}
  \item try to differentiate $\mathbf{S}^{T}\mathbf{S} = 0$
\end{itemize}
\end{frame}

\begin{frame}{The velocity-form Lagrangian --- everything, electrons included}
\setstage{2}
Take the internal energy of Sec.~2 in \emph{velocity} form (before the Legendre transform, eq.~A.12) and body-fix it: substitute the co-rotating velocities (3.3)
$\dot{\mathbf R}'_j=\mathbf S(\boldsymbol\omega\!\times\!\bar{\mathbf R}_j+\dot{\bar{\mathbf R}}_j)$, $\dot{\mathbf r}'_i=\mathbf S(\boldsymbol\omega\!\times\!\bar{\mathbf r}_i+\dot{\bar{\mathbf r}}_i)$, using $|\mathbf S\mathbf v|^2=|\mathbf v|^2$. \alert{Every particle stays in velocity form:}
\begin{equation*}
\boxed{\;\mathcal L=\tfrac12 M_{\mathrm{tot}}|\dot{\mathbf R}_{\mathrm{cm}}|^2
+\tfrac12\sum_j M_j|\boldsymbol\omega\!\times\!\bar{\mathbf R}_j+\dot{\bar{\mathbf R}}_j|^2
+\tfrac{m_e}{2}\sum_i|\boldsymbol\omega\!\times\!\bar{\mathbf r}_i+\dot{\bar{\mathbf r}}_i|^2
-\frac{m_e^2}{2M_{\mathrm{tot}}}|\boldsymbol\omega\!\times\!\bar{\mathbf r}+\dot{\bar{\mathbf r}}|^2-V\;}\tag{3.7$'$}
\end{equation*}
with $\bar{\mathbf r}\equiv\sum_i\bar{\mathbf r}_i$. The potential is now manifestly $\mathbf S$-independent (3.8). We carry the electrons in velocity form all the way --- $\mathbf I_e$, $\mathbf I_X$ and all.
\end{frame}

\begin{frame}{The vibrational term: $\dot{\bar{\mathbf R}}_j\to\dot Q_b$}
\setstage{2}
Expanding each square $|\boldsymbol\omega\!\times\!\mathbf a+\dot{\mathbf a}|^2$ leaves a pure-velocity (\alert{vibrational}) piece $|\dot{\mathbf a}|^2$. For the nuclei it collapses cleanly: the mode vectors are \emph{constant} [$\partial\bar{\mathbf R}_j/\partial Q_b=\mathbf l_{jb}/\sqrt{M_j}$ from (3.1a)], so
\begin{equation*}
\dot{\bar{\mathbf R}}_j=\frac{1}{\sqrt{M_j}}\sum_b\mathbf l_{jb}\,\dot Q_b .
\end{equation*}
The $M_j$ cancels and the orthonormality $\sum_j\mathbf l_{jb}\!\cdot\!\mathbf l_{jc}=\delta_{bc}$ (3.1a) collapses the double sum:
\begin{equation*}
\boxed{\;T_{\mathrm{vib}}^{N}\equiv\tfrac12\sum_j M_j|\dot{\bar{\mathbf R}}_j|^2
=\tfrac12\sum_{b,c}\Bigl(\underbrace{\textstyle\sum_j\mathbf l_{jb}\!\cdot\!\mathbf l_{jc}}_{=\,\delta_{bc}}\Bigr)\dot Q_b\dot Q_c
=\tfrac12\sum_b\dot Q_b^2 .\;}
\end{equation*}
\alert{The vibrational metric is the identity --- no Wilson $G$-matrix.} The electron kinetic and mass-polarization pieces stay in velocity form; every symbol is collected on the next slide.
\end{frame}

\begin{frame}{Expand every squared term $\Rightarrow$ the full inertia}
\setstage{2}
The vector identity (any $\mathbf a,\dot{\mathbf a},\boldsymbol\omega$)
\begin{equation*}
\tfrac{m}{2}|\boldsymbol{\omega}\!\times\!\mathbf{a}+\dot{\mathbf{a}}|^{2}
=\underbrace{\tfrac{m}{2}\boldsymbol{\omega}^{T}(a^{2}\mathbb{1}-\mathbf{a}\mathbf{a}^{T})\boldsymbol{\omega}}_{\text{rotational}}
+\underbrace{m\,\boldsymbol{\omega}\!\cdot\!(\mathbf{a}\!\times\!\dot{\mathbf{a}})}_{\text{Coriolis}}
+\underbrace{\tfrac{m}{2}|\dot{\mathbf{a}}|^{2}}_{\text{vibrational}} \tag{3.9c}
\end{equation*}
applied to \emph{nuclei, electrons, and the mass-polarization term} gives
\begin{equation*}
\boxed{\;\mathcal L=\tfrac12 M_{\mathrm{tot}}|\dot{\mathbf R}_{\mathrm{cm}}|^2
+\tfrac12\boldsymbol\omega^{T}\underbrace{(\mathbf I_N+\mathbf I_e-\mathbf I_X)}_{\text{full inertia }\mathbf I}\boldsymbol\omega
+\boldsymbol\omega\!\cdot\!(\mathbf L_N^{\mathrm{vib}}+\mathbf L_e-\mathbf L_X)
+\tfrac12\sum_b\dot Q_b^2+T_e^{\mathrm{BF}}-T_X^{\mathrm{mp}}-V\;}\tag{3.13$'$}
\end{equation*}
\scriptsize\textbf{Definitions.} \emph{Inertia} (3.9,3.10): $\mathbf I_N=\sum_jM_j(\bar R_j^2\mathbb 1-\bar{\mathbf R}_j\bar{\mathbf R}_j^{T})$, $\mathbf I_e=m_e\sum_i(\bar r_i^2\mathbb 1-\bar{\mathbf r}_i\bar{\mathbf r}_i^{T})$, $\mathbf I_X=\tfrac{m_e^2}{M_{\mathrm{tot}}}(\bar r^2\mathbb 1-\bar{\mathbf r}\bar{\mathbf r}^{T})$. \emph{Angular momenta} (3.11): $\mathbf L_N^{\mathrm{vib}}=\sum_jM_j\bar{\mathbf R}_j\!\times\!\dot{\bar{\mathbf R}}_j$, $\mathbf L_e=m_e\sum_i\bar{\mathbf r}_i\!\times\!\dot{\bar{\mathbf r}}_i$, $\mathbf L_X=\tfrac{m_e^2}{M_{\mathrm{tot}}}\bar{\mathbf r}\!\times\!\dot{\bar{\mathbf r}}$. \emph{Kinetic pieces} (3.12): $T_e^{\mathrm{BF}}=\tfrac{m_e}{2}\sum_i|\dot{\bar{\mathbf r}}_i|^2$, $T_X^{\mathrm{mp}}=\tfrac{m_e^2}{2M_{\mathrm{tot}}}|\dot{\bar{\mathbf r}}|^2$ (with $\tfrac12\sum_b\dot Q_b^2=T_{\mathrm{vib}}^{N}$ from the previous slide).\normalsize
\end{frame}

\begin{frame}{Total, nuclear, and canonical electronic angular momenta}
\setstage{2}
Read the momenta off $\mathcal L$ (3.13$'$):
\begin{equation*}
\mathbf J\equiv\frac{\partial \mathcal L}{\partial\boldsymbol\omega}
=\underbrace{(\mathbf I_N+\mathbf I_e-\mathbf I_X)\boldsymbol\omega+\mathbf L_N^{\mathrm{vib}}+\mathbf L_e-\mathbf L_X}_{\text{total ang.\ mom.}},
\qquad
\mathbf J_N\equiv\mathbf I_N\boldsymbol\omega+\mathbf L_N^{\mathrm{vib}}\ \ (\text{nuclear}). \tag{3.16, 3.15}
\end{equation*}
The \alert{canonical electronic angular momentum} is built from the electron pair $(\bar{\mathbf r}_i,\bar{\mathbf p}_i)$:
\begin{equation*}
\mathbf L_{\mathrm{elec}}=\sum_i\bar{\mathbf r}_i\!\times\!\bar{\mathbf p}_i,\qquad
\bar{\mathbf p}_i\equiv\frac{\partial \mathcal L}{\partial\dot{\bar{\mathbf r}}_i}
=m_e(\boldsymbol\omega\!\times\!\bar{\mathbf r}_i+\dot{\bar{\mathbf r}}_i)-\frac{m_e^2}{M_{\mathrm{tot}}}(\boldsymbol\omega\!\times\!\bar{\mathbf r}+\dot{\bar{\mathbf r}}). \tag{3.14}
\end{equation*}
Vibrational momentum, with the \alert{Coriolis vectors} $\boldsymbol\sigma_b$:
\begin{equation*}
P_b=\frac{\partial \mathcal L}{\partial\dot Q_b}=\boldsymbol\omega\!\cdot\!\boldsymbol\sigma_b+\dot Q_b,\qquad
\boldsymbol\sigma_b=\sum_a\boldsymbol\zeta_{ab}Q_a,\quad \boldsymbol{\zeta}_{ab}=\sum_j\mathbf l_{ja}\!\times\!\mathbf l_{jb}. \tag{3.17a,b}
\end{equation*}
\end{frame}

\begin{frame}{Proof 1: $\mathbf L_{\mathrm{elec}}$ \emph{is} the electron $+$ mass-polarization ang.\ mom.}
\setstage{2}
Substitute $\bar{\mathbf p}_i$ (3.14) into $\mathbf L_{\mathrm{elec}}=\sum_i\bar{\mathbf r}_i\!\times\!\bar{\mathbf p}_i$ and use BAC--CAB on the two $\boldsymbol\omega$-terms,
\begin{equation*}
\mathbf a\!\times\!(\boldsymbol\omega\!\times\!\mathbf a)=(a^2\mathbb 1-\mathbf a\mathbf a^{T})\boldsymbol\omega :
\end{equation*}
\begin{equation*}
\boxed{\;\mathbf L_{\mathrm{elec}}=(\mathbf I_e-\mathbf I_X)\boldsymbol\omega+(\mathbf L_e-\mathbf L_X).\;}\tag{3.18}
\end{equation*}
\begin{block}{What this says}
The canonical electronic angular momentum is \emph{exactly} the electron and mass-polarization inertia/Coriolis pieces of $\mathcal L$ (3.13$'$). Therefore the total splits cleanly,
\[
\mathbf J=\underbrace{\mathbf I_N\boldsymbol\omega+\mathbf L_N^{\mathrm{vib}}}_{\mathbf J_N}+\underbrace{(\mathbf I_e-\mathbf I_X)\boldsymbol\omega+(\mathbf L_e-\mathbf L_X)}_{\mathbf L_{\mathrm{elec}}}=\mathbf J_N+\mathbf L_{\mathrm{elec}}. \tag{3.16}
\]
\end{block}
\end{frame}

\begin{frame}{Proof 2: the electron sector recovers $H_e$}
\setstage{2}
Write the body-frame electron velocity $\mathbf w_i\equiv\boldsymbol\omega\!\times\!\bar{\mathbf r}_i+\dot{\bar{\mathbf r}}_i$. The electron $+$ mass-polarization part of $\mathcal L$ (3.7$'$) is
\begin{equation*}
\mathcal L_e^{\mathrm{vel}}=\tfrac{m_e}{2}\sum_i|\mathbf w_i|^2-\frac{m_e^2}{2M_{\mathrm{tot}}}\Bigl|\sum_i\mathbf w_i\Bigr|^2,
\qquad
\bar{\mathbf p}_i=m_e\mathbf w_i-\frac{m_e^2}{M_{\mathrm{tot}}}\sum_l\mathbf w_l,
\end{equation*}
\alert{identical in structure to the Sec.~2 velocity form (A.12) with $\dot{\mathbf r}'_i\to\mathbf w_i$.} The same mass-matrix inversion $\mathbf A\to\mathbf A^{-1}$ (A.25--A.26) gives
\begin{equation*}
\boxed{\;\sum_i\bar{\mathbf p}_i\!\cdot\!\mathbf w_i-\mathcal L_e^{\mathrm{vel}}
=\sum_i\frac{\bar{\mathbf p}_i^{\,2}}{2m_e}+\frac{1}{2M_N}\Bigl(\sum_i\bar{\mathbf p}_i\Bigr)^{2}\equiv H_e.\;}
\end{equation*}
\begin{block}{Why $\boldsymbol\omega$ does not spoil it}
In $\sum p\dot q$, use $\dot{\bar{\mathbf r}}_i=\mathbf w_i-\boldsymbol\omega\!\times\!\bar{\mathbf r}_i$:\ \ $\sum_i\bar{\mathbf p}_i\!\cdot\!\dot{\bar{\mathbf r}}_i=\sum_i\bar{\mathbf p}_i\!\cdot\!\mathbf w_i-\boldsymbol\omega\!\cdot\!\mathbf L_{\mathrm{elec}}$. The $-\boldsymbol\omega\!\cdot\!\mathbf L_{\mathrm{elec}}$ cancels the electronic part of $\mathbf J\!\cdot\!\boldsymbol\omega$, leaving $\mathbf J_N\!\cdot\!\boldsymbol\omega$ for rotation and the clean $H_e$ for electrons.
\end{block}
\end{frame}

\begin{frame}{The key identity}
\setstage{2}
Combining (3.16) and (3.18), the electronic and mass-polarization inertias cancel between $\mathbf J$ and $\mathbf L_{\mathrm{elec}}$:
\begin{equation*}
\boxed{\;\mathbf{J}-\mathbf{L}_{\mathrm{elec}}=\mathbf{I}_N(Q)\,\boldsymbol{\omega}+\mathbf{L}_N^{\mathrm{vib}}.\;} \tag{3.19}
\end{equation*}
\begin{block}{The algebraic miracle}
The full inertia $\mathbf I_N+\mathbf I_e-\mathbf I_X$ appeared in $\mathcal L$, but $\mathbf I_e,\mathbf I_X$ drop out of the right-hand side: the surviving rotational kernel is the \emph{nuclear} inertia $\mathbf I_N$ \textbf{alone}. All electronic $+$ mass-polarization content sits in the kinematic shift $\mathbf J\to\mathbf J-\mathbf L_{\mathrm{elec}}$, and the electron sector is the clean scalar $H_e$ (Proof 2).
\end{block}
\end{frame}

\begin{frame}{Watson reciprocal inertia}
\setstage{2}
Invert (3.19) for $\boldsymbol\omega$: write $\mathbf L_N^{\mathrm{vib}}=\boldsymbol\pi-\boldsymbol\sigma\boldsymbol\sigma^{\mathsf T}\boldsymbol\omega$ with
\begin{equation*}
\boldsymbol{\pi}=\sum_b\boldsymbol{\sigma}_bP_b\ \ (\text{field-free vibrational ang.\ mom.}),\qquad
\boldsymbol{\sigma}\boldsymbol{\sigma}^{\mathsf T}=\sum_b\boldsymbol{\sigma}_b\boldsymbol{\sigma}_b^{\mathsf T}. \tag{3.20a}
\end{equation*}
Then $\mathbf J-\mathbf L_{\mathrm{elec}}=(\mathbf I_N-\boldsymbol\sigma\boldsymbol\sigma^{\mathsf T})\boldsymbol\omega+\boldsymbol\pi$, so
\begin{equation*}
\boxed{\;\boldsymbol{\omega}=\boldsymbol{\mu}\bigl(\mathbf{J}-\mathbf{L}_{\mathrm{elec}}-\boldsymbol{\pi}\bigr),\qquad
\boldsymbol{\mu}\equiv(\mathbf{I}_N-\boldsymbol{\sigma}\boldsymbol{\sigma}^{\mathsf T})^{-1}.\;}\tag{3.20}
\end{equation*}
$\boldsymbol\mu$ is the \alert{Watson reciprocal-inertia tensor} --- built from \emph{nuclear} quantities only, with no Wilson $G$-matrix.
\end{frame}

\begin{frame}{The exact classical molecular Hamiltonian}
\setstage{2}
Legendre back. The rovibrational block gives $\tfrac12(\mathbf J-\mathbf L_{\mathrm{elec}}-\boldsymbol\pi)^{\mathsf T}\boldsymbol\mu(\mathbf J-\mathbf L_{\mathrm{elec}}-\boldsymbol\pi)+\tfrac12\sum_bP_b^2$; the electron sector gives $H_e$ (Proof 2):
\begin{equation*}
\boxed{
\begin{aligned}
H_{\mathrm{int}}=\;&\tfrac12\bigl(\mathbf{J}-\mathbf{L}_{\mathrm{elec}}-\boldsymbol{\pi}\bigr)^{\mathsf T}\boldsymbol{\mu}(Q)\bigl(\mathbf{J}-\mathbf{L}_{\mathrm{elec}}-\boldsymbol{\pi}\bigr)+\tfrac12\!\sum_bP_b^2\\
&+\sum_i\frac{\bar{\mathbf p}_i^{\,2}}{2m_e}+\frac{1}{2M_N}\Bigl(\sum_i\bar{\mathbf p}_i\Bigr)^{2}+V_{NN}+V_{Ne}+V_{ee}.
\end{aligned}}\tag{3.22}
\end{equation*}
\centering Exact, classical, in BF + Eckart coordinates. \alert{Now we quantize.}
\end{frame}

\section{The quantization problem}

\begin{frame}{Primer: wavefunctions live in a space, $L^2$}
\setstage{3}
In quantum mechanics a state is a \alert{wavefunction} $\psi(q)$ --- and wavefunctions add and scale like vectors, so they form a vector space. To measure ``length'' and ``angle'' we need an \alert{inner product}; for two states $\varphi,\psi$,
\begin{equation*}
\langle\varphi|\psi\rangle \;\equiv\; \int \varphi^{*}(q)\,\psi(q)\;dq
\qquad(\text{Dirac ``bra--ket'' notation}).
\end{equation*}
The squared length is the \alert{norm}, and a physical state is \alert{normalized}:
\begin{equation*}
\|\psi\|^2=\langle\psi|\psi\rangle=\int|\psi(q)|^2\,dq \;=\;1
\qquad(\text{total probability}=1).
\end{equation*}
\begin{block}{What is $L^2$?}
$L^2$ is the space of \alert{square-integrable} functions --- those with finite $\int|\psi|^2<\infty$, so the norm makes sense. Operators like $\hat H$ act on this space, and observables must be \alert{Hermitian}, $\hat O^{\dagger}=\hat O$, to have real eigenvalues.
\end{block}
\end{frame}

\begin{frame}{Primer: the \emph{measure} --- what you integrate against}
\setstage{3}
The inner product needs a rule for ``how much volume'' each point carries --- the \alert{measure}. In plain Cartesian coordinates it is $dq=dx\,dy\,dz$, but after a change of variables it picks up the \alert{Jacobian}:
\begin{equation*}
\int f\,dx\,dy\,dz=\int f\;\underbrace{r^2\sin\theta}_{\text{Jacobian}}\,dr\,d\theta\,d\phi
\quad\Longrightarrow\quad
\langle\varphi|\psi\rangle=\int\varphi^{*}\psi\;\underbrace{w(q)}_{\text{measure}}\,d^nq .
\end{equation*}
Normalization is always against this measure: $\int|\psi|^2\,w\,d^nq=1$. We write the space as $L^2(w\,d^nq)$.
\begin{alertblock}{Why it matters here}
\alert{Hermiticity depends on the measure.} On a curved measure $-i\hbar\partial$ is \emph{not} Hermitian (next slides show this for $w=r^2\sin\theta$). Getting the operator right on the correct measure is exactly the quantization problem --- and its cure is the Podolsky rule. For the molecule the measure will be $\sqrt g\,d^nq$ (later reshaped to the flat $d\mu_W$).
\end{alertblock}
\end{frame}

\begin{frame}{A worked example: one particle in spherical coordinates}
\setstage{3}
Take a single free particle of mass $m$. Cartesian: $\mathcal L=\tfrac12 m(\dot x^2+\dot y^2+\dot z^2)$. In spherical coordinates $(r,\theta,\phi)$,
\begin{equation*}
\mathcal L=\tfrac12 m\bigl(\dot r^2+r^2\dot\theta^2+r^2\sin^2\theta\,\dot\phi^2\bigr).
\end{equation*}
Canonical momenta $p_a=\partial\mathcal L/\partial\dot q^a$:
\begin{equation*}
p_r=m\dot r,\qquad p_\theta=mr^2\dot\theta,\qquad p_\phi=mr^2\sin^2\theta\,\dot\phi .
\end{equation*}
Legendre transform $\Rightarrow$ the classical Hamiltonian
\begin{equation*}
\boxed{\;H=\frac{p_r^2}{2m}+\frac{p_\theta^2}{2mr^2}+\frac{p_\phi^2}{2mr^2\sin^2\theta}.\;}
\end{equation*}
The coordinates are \emph{curvilinear}: $p_\theta,p_\phi$ are not mass$\times$velocity, and the coefficients depend on $r,\theta$.
\end{frame}

\begin{frame}{Naive quantization gives the wrong operator}
\setstage{3}
Substitute $p_a\to-i\hbar\,\partial_a$ into $H$:
\begin{equation*}
\hat H_{\mathrm{naive}}=-\frac{\hbar^2}{2m}\Bigl[\partial_r^2+\frac{1}{r^2}\partial_\theta^2+\frac{1}{r^2\sin^2\theta}\partial_\phi^2\Bigr].
\end{equation*}
But the true kinetic operator is $-\tfrac{\hbar^2}{2m}\nabla^2$, with the familiar spherical Laplacian
\begin{equation*}
\nabla^2=\underbrace{\frac{1}{r^2}\partial_r(r^2\partial_r)}_{\partial_r^2+\frac{2}{r}\partial_r}
+\underbrace{\frac{1}{r^2\sin\theta}\partial_\theta(\sin\theta\,\partial_\theta)}_{\frac{1}{r^2}(\partial_\theta^2+\cot\theta\,\partial_\theta)}
+\frac{1}{r^2\sin^2\theta}\partial_\phi^2 .
\end{equation*}
\alert{Naive quantization is missing the first-derivative terms $\tfrac{2}{r}\partial_r$ and $\tfrac{\cot\theta}{r^2}\partial_\theta$.} Where did they go?
\end{frame}

\begin{frame}{Why it fails: $-i\hbar\partial$ is not Hermitian on the curved measure}
\setstage{3}
The physical inner product carries the volume Jacobian:
\begin{equation*}
\langle\varphi|\psi\rangle=\int\varphi^{*}\psi\;\underbrace{r^2\sin\theta}_{\text{Jacobian}}\,dr\,d\theta\,d\phi .
\end{equation*}
Is $\hat p_r=-i\hbar\partial_r$ Hermitian here? Integrate by parts in $r$ (boundary terms vanish):
\begin{equation*}
\int\varphi^{*}(-i\hbar\partial_r\psi)\,r^2\,dr
=\int(-i\hbar\partial_r\varphi)^{*}\psi\,r^2\,dr
-i\hbar\int\varphi^{*}\psi\,\underbrace{(\partial_r r^2)}_{=\,(2/r)\,r^2}dr,
\end{equation*}
so
\begin{equation*}
\boxed{\;\hat p_r^{\dagger}=-i\hbar\Bigl(\partial_r+\frac{2}{r}\Bigr)\neq\hat p_r.\;}
\end{equation*}
Not Hermitian $\Rightarrow$ $\hat p_r^{\,2}$ built this way isn't either $\Rightarrow$ no guaranteed real spectrum. The naive recipe \alert{ignored the measure}. (Same for $\theta$: the $\cot\theta$ term.)
\end{frame}

\begin{frame}{The Podolsky / Laplace--Beltrami rule}
\setstage{3}
Cure: read the kinetic energy as a quadratic form, $T=\tfrac12 g_{ab}(q)\dot q^a\dot q^b=\tfrac12 g^{ab}(q)\,p_ap_b$, which \emph{defines} the metric $g_{ab}$ and the measure $\sqrt g\,d^nq$. The operator Hermitian on $L^2(\sqrt g\,d^nq)$ \emph{by construction} is
\begin{equation*}
\boxed{\;\hat T=-\frac{\hbar^2}{2}\,\frac{1}{\sqrt g}\,\partial_a\!\bigl(\sqrt g\,g^{ab}\,\partial_b\bigr),\qquad g=\det g_{ab}.\;}\tag{4.5}
\end{equation*}
One integration by parts gives the manifestly symmetric, positive form
\begin{equation*}
\int\varphi^{*}\hat T\psi\,\sqrt g\,d^nq=\frac{\hbar^2}{2}\int(\partial_a\varphi)^{*}g^{ab}(\partial_b\psi)\,\sqrt g\,d^nq . \tag{4.6}
\end{equation*}
The $\sqrt g$ factors are exactly the measure the naive recipe forgot; for flat Cartesian $g_{ab}=m\delta_{ab}$ it is $-\tfrac{\hbar^2}{2m}\nabla^2$.
\end{frame}

\begin{frame}{Podolsky recovers the correct Hamiltonian}
\setstage{3}
For the spherical particle, $\mathcal L$ gives $g_{ab}=\mathrm{diag}(m,\,mr^2,\,mr^2\sin^2\theta)$, so
\begin{equation*}
\sqrt g=m^{3/2}r^2\sin\theta,\qquad g^{ab}=\mathrm{diag}\Bigl(\tfrac1m,\ \tfrac1{mr^2},\ \tfrac1{mr^2\sin^2\theta}\Bigr).
\end{equation*}
Insert into (4.5); the constant $m^{3/2}$ cancels between $\sqrt g$ and $1/\sqrt g$:
\begin{equation*}
\hat T=-\frac{\hbar^2}{2m}\Bigl[\frac{1}{r^2}\partial_r(r^2\partial_r)+\frac{1}{r^2\sin\theta}\partial_\theta(\sin\theta\,\partial_\theta)+\frac{1}{r^2\sin^2\theta}\partial_\phi^2\Bigr]=-\frac{\hbar^2}{2m}\nabla^2. \ \checkmark
\end{equation*}
The $\tfrac2r\partial_r$ and $\cot\theta\,\partial_\theta$ are \alert{restored automatically} --- by the measure, not by hand.
\begin{block}{The molecule is the same story}
Curvilinear $\{q^A\}=\{\phi,\theta,\chi,Q_1,\dots,Q_f\}$, $T=\tfrac12 g^{AB}p_Ap_B$, curved measure $\sqrt g\,d^nq$. We now build the three objects --- metric $g_{AB}$, determinant $g$, inverse $g^{AB}$ --- explicitly, then apply (4.5). \emph{Next section.}
\end{block}
\end{frame}

\begin{frame}{Recipe: Podolsky / Laplace--Beltrami quantization}
\setstage{3}
\small
\begin{columns}[T]
\begin{column}{0.58\textwidth}
\textbf{General recipe} --- curvilinear $\{q^i\}$:
\begin{enumerate}\setlength{\itemsep}{0pt}
\item \textbf{Metric} $g_{ij}$: read off $T=\tfrac12 g_{ij}\dot q^i\dot q^j$, or $g_{ij}=\sum_a m_a\,(\partial\mathbf r_a/\partial q^i)\!\cdot\!(\partial\mathbf r_a/\partial q^j)$.
\item \textbf{Determinant \& measure}: $g=\det g_{ij}$ $\Rightarrow$ Hilbert space $L^2(\sqrt g\,d^nq)$.
\item \textbf{Inverse} $g^{ij}=(g_{ij})^{-1}$.
\item \textbf{Podolsky operator} (Hermitian on $L^2(\sqrt g\,d^nq)$ by construction):
\[ \hat T=-\frac{\hbar^2}{2}\,\frac{1}{\sqrt g}\,\partial_i\!\bigl(\sqrt g\,g^{ij}\partial_j\bigr); \]
$V$ and coordinate functions $\to$ multiplication.
\item \textbf{(Optional) rescale} $\psi\to g^{1/4}\psi$ to a flatter measure; leftover $c$-number $=$ \alert{pseudopotential}.
\end{enumerate}
\end{column}
\begin{column}{0.42\textwidth}
\footnotesize
\begin{block}{Instance: free particle in $(r,\theta,\phi)$}
\begin{enumerate}\setlength{\itemsep}{1pt}
\item $g_{ij}=\mathrm{diag}(m,\,mr^2,\,mr^2\sin^2\theta)$
\item $\sqrt g=m^{3/2}r^2\sin\theta$
\item $g^{ij}$: reciprocals of $g_{ii}$
\item $\hat T=-\dfrac{\hbar^2}{2m}\nabla^2$ \checkmark
\item flat $\Rightarrow$ no pseudopotential
\end{enumerate}
\end{block}

\end{column}
\end{columns}
\end{frame}

\section{The ro-vibrational geometry}

\begin{frame}{Euler-angle kinematics, explicitly}
\setstage{3}
$zyz$ convention $\mathbf S=R_z(\phi)R_y(\theta)R_z(\chi)$ (4.7). Differentiating $\mathbf S$ gives $\boldsymbol\omega=\mathsf E(\Omega)\dot{\boldsymbol\Omega}$ and, solving the linear system, $\dot{\boldsymbol\Omega}=\mathsf E^{-1}\boldsymbol\omega$:
\begin{equation*}
\mathsf E=\begin{pmatrix}-\sin\theta\cos\chi&\sin\chi&0\\ \ \ \sin\theta\sin\chi&\cos\chi&0\\ \ \ \cos\theta&0&1\end{pmatrix},\qquad
\mathsf E^{-1}=\begin{pmatrix}-\frac{\cos\chi}{\sin\theta}&\frac{\sin\chi}{\sin\theta}&0\\[3pt] \ \ \sin\chi&\cos\chi&0\\[3pt] \ \ \frac{\cos\theta\cos\chi}{\sin\theta}&-\frac{\cos\theta\sin\chi}{\sin\theta}&1\end{pmatrix} \tag{4.10, 4.12}
\end{equation*}
with
\begin{equation*}
\det\mathsf E=-\sin\theta . \tag{4.11}
\end{equation*}
\footnotesize The $1/\sin\theta$ entries are the Euler coordinate singularity at the poles; they are compensated by $|\det\mathsf E|=\sin\theta$ in the volume element. \alert{(4.10)--(4.12) are the only Euler-angle facts used from here on.}
\end{frame}

\begin{frame}{The ro-vibrational metric tensor}
\setstage{3}
The classical BF kinetic energy $2T_{\mathrm{rv}}=\boldsymbol\omega^{T}\mathbf I_N\boldsymbol\omega+2\boldsymbol\omega^{T}\!\sum_b\boldsymbol\sigma_b\dot Q_b+\sum_b\dot Q_b^2$ (4.13), re-expressed in the \emph{coordinate} velocities via $\boldsymbol\omega=\mathsf E\dot{\boldsymbol\Omega}$, defines the metric:
\begin{equation*}
\boxed{\;g_{AB}=
\begin{pmatrix}
\mathsf E^{T}\mathbf I_N\,\mathsf E & \mathsf E^{T}\boldsymbol\sigma\\[3pt]
\boldsymbol\sigma^{T}\mathsf E & \mathbb 1_f
\end{pmatrix}\;}\tag{4.15}
\end{equation*}
\begin{itemize}
  \item Euler--Euler block: inertia pulled back through the kinematic matrix.
  \item Euler--mode block: Coriolis coupling $\boldsymbol\sigma_b=\sum_a\boldsymbol\zeta_{ab}Q_a$ (constant $\boldsymbol\zeta$'s).
  \item mode--mode block: \emph{identity} --- the gift of mass-weighted rectilinear normal coordinates.
\end{itemize}
Every entry is an explicit function of $(\theta,\chi,Q)$.
\end{frame}

\begin{frame}{Electrons: flat, constant, decoupled from the measure}
\setstage{3}
Including electron velocities, the full mass matrix has a \emph{constant} electron block $\mathsf D$. Eliminating it (Schur complement $=$ completing the square) removes all $\boldsymbol\omega$-dependence:
\begin{equation*}
\mathsf C\mathsf D^{-1}\mathsf C^{\mathsf T}=\mathbf I_e-\mathbf I_X
\;\Longrightarrow\;
\det\mathsf M=\underbrace{\det\mathsf D}_{\text{const}}\cdot\det\!\begin{pmatrix}\mathbf I_N&\boldsymbol\sigma\\ \boldsymbol\sigma^{\mathsf T}&\mathbb 1_f\end{pmatrix} \tag{4.17}
\end{equation*}
--- the same $\mathbf I_e/\mathbf I_X$ cancellation as the classical cross-check of \S\,3.6.
\begin{itemize}
  \item Electron measure is the flat $\prod_i d^3\bar{\mathbf r}_i$; $\hat{\bar{\mathbf p}}_i=-i\hbar\nabla_{\bar{\mathbf r}_i}$ is Hermitian on it.
  \item $\hat L_{\mathrm{elec},\alpha}$ is Hermitian \emph{without symmetrization}: its vector field is divergence-free,
  $\sum_{i\gamma}\partial_{\bar r_{i\gamma}}(\epsilon_{\alpha\beta\gamma}\bar r_{i\beta})=0$ (4.18).
\end{itemize}
\alert{The entire quantization problem lives in the ro-vibrational sector}; electrons enter only via the flat kinetic operator and the shift $\hat J_\alpha\to\hat J_\alpha-\hat L_{\mathrm{elec},\alpha}$.
\end{frame}

\begin{frame}{Determinant and volume element}
\setstage{3}
Block row reduction gives $\det\bigl(\begin{smallmatrix}\mathsf a&\mathsf b\\ \mathsf b^{T}&\mathbb 1\end{smallmatrix}\bigr)=\det(\mathsf a-\mathsf b\mathsf b^{T})$ (4.19). Applied to (4.15):
\begin{equation*}
\boxed{\;\det g_{AB}=\sin^2\!\theta\;\gamma(Q),\qquad
\gamma\equiv\det\mathbf I',\qquad
\mathbf I'\equiv\mathbf I_N-\boldsymbol\sigma\boldsymbol\sigma^{T}.\;}\tag{4.20}
\end{equation*}
\begin{block}{The Watson-modified inertia appears by itself}
The determinant is governed by $\mathbf I'=\mathbf I_N-\boldsymbol\sigma\boldsymbol\sigma^{T}$ --- \emph{not} the bare $\mathbf I_N$: the Coriolis coupling reduces the rotational determinant. It is the same $\mathbf I'$ whose inverse $\boldsymbol\mu$ is the classical rotational kernel (3.20).
\end{block}
\begin{equation*}
\sqrt{g}\,d^{\,3+f}q=\underbrace{\sin\theta\,d\phi\,d\theta\,d\chi}_{\text{Haar measure on }SO(3)}\cdot\underbrace{\sqrt{\gamma(Q)}\,\textstyle\prod_bdQ_b}_{\text{rot.--vib.\ Jacobian}}\cdot\underbrace{\textstyle\prod_id^3\bar{\mathbf r}_i}_{\text{flat}} \tag{4.21}
\end{equation*}
\end{frame}

\begin{frame}{The inverse metric}
\setstage{3}
Claim, then \emph{verify} by block multiplication using only $\boldsymbol\mu\mathbf I'=\mathbb 1$ [(4.23)]:
\begin{equation*}
\boxed{\;g^{AB}=
\begin{pmatrix}
\mathsf E^{-1}\boldsymbol\mu\,\mathsf E^{-T} & -\mathsf E^{-1}\boldsymbol\mu\,\boldsymbol\sigma\\[3pt]
-\boldsymbol\sigma^{T}\boldsymbol\mu\,\mathsf E^{-T} & \mathbb 1_f+\boldsymbol\sigma^{T}\boldsymbol\mu\,\boldsymbol\sigma
\end{pmatrix},
\qquad \boldsymbol\mu=(\mathbf I_N-\boldsymbol\sigma\boldsymbol\sigma^{T})^{-1}.\;}\tag{4.22}
\end{equation*}
\begin{block}{Two inversions, one computation}
The Watson reciprocal inertia $\boldsymbol\mu$ --- introduced classically by inverting the momentum--velocity relation (3.20) --- reappears as the geometric inverse of the metric.
\end{block}
\footnotesize e.g.\ top-left: $\mathsf E^{T}\mathbf I_N\mathsf E\cdot\mathsf E^{-1}\boldsymbol\mu\mathsf E^{-T}-\mathsf E^{T}\boldsymbol\sigma\cdot\boldsymbol\sigma^{T}\boldsymbol\mu\mathsf E^{-T}=\mathsf E^{T}(\mathbf I_N-\boldsymbol\sigma\boldsymbol\sigma^{T})\boldsymbol\mu\mathsf E^{-T}=\mathbb 1_3$. $\checkmark$
\end{frame}

\section{Quantizing rotation \& Coriolis coupling}

\begin{frame}{Why the $\hat J_\alpha$ look unlike the usual angular momentum}
\setstage{3}
The rotational operators below differ from the $\hat L$ of introductory QM for \emph{three} reasons:
\begin{enumerate}\setlength{\itemsep}{2pt}
\item \textbf{Three angles, not two.} A rigid body needs $(\phi,\theta,\chi)$, so besides $\partial_\phi,\partial_\theta$ there is a third generator $\hat J_3=-i\hbar\partial_\chi$, and the three mix.
\item \textbf{Body-fixed, not space-fixed.} They are components along the \emph{molecule's} axes, $\hat J_\alpha^{\mathrm{BF}}=S_{\beta\alpha}(\Omega)\hat J_\beta^{\mathrm{SF}}$; since $\mathbf S$ depends on the angles, their commutator carries the \alert{anomalous sign} $[\hat J_\alpha,\hat J_\beta]=-i\hbar\epsilon_{\alpha\beta\gamma}\hat J_\gamma$ --- which we \emph{never use}.
\item \textbf{Convention-dependent form.} The explicit trig coefficients are fixed by our $zyz$ choice $\mathbf S=R_z(\phi)R_y(\theta)R_z(\chi)$; another convention relabels them ($zyz$ is itself standard --- Wigner, Zare).
\end{enumerate}
\centering\itshape Nothing is postulated --- the operators fall out of the Laplace--Beltrami expansion next.
\end{frame}

\begin{frame}{Laplace--Beltrami on the rotational block $\Rightarrow$ $\hat J_\alpha$}
\setstage{3}
Put the measure $\sqrt g=\sin\theta\sqrt\gamma$ and the rotational inverse-metric block $g^{\Omega\Omega}_{st}=(\mathsf E^{-1})_{s\alpha}\mu_{\alpha\beta}(\mathsf E^{-1})_{t\beta}$ into the Podolsky/LB operator (4.5). Since $\mu,\sqrt\gamma$ depend on $Q$ only, they pass through $\partial_\Omega$:
\begin{equation*}
\hat T_{\mathrm{rot}}=-\frac{\hbar^2}{2}\,\mu_{\alpha\beta}\,\frac{1}{\sin\theta}\,\partial_s\!\bigl(\sin\theta\,(\mathsf E^{-1})_{s\alpha}\,(\mathsf E^{-1})_{t\beta}\,\partial_t\bigr).
\end{equation*}
The combination $(\mathsf E^{-1})_{s\alpha}\partial_{\Omega_s}$ appears twice --- \alert{name it}:
\begin{equation*}
\boxed{\;\hat J_\alpha\equiv-i\hbar\!\sum_s(\mathsf E^{-1})_{s\alpha}\partial_{\Omega_s}:\ 
\begin{aligned}
\hat J_1&=-i\hbar[-\tfrac{\cos\chi}{\sin\theta}\partial_\phi+\sin\chi\,\partial_\theta+\tfrac{\cos\theta\cos\chi}{\sin\theta}\partial_\chi],\\
\hat J_2&=-i\hbar[\ \tfrac{\sin\chi}{\sin\theta}\partial_\phi+\cos\chi\,\partial_\theta-\tfrac{\cos\theta\sin\chi}{\sin\theta}\partial_\chi],\\
\hat J_3&=-i\hbar\,\partial_\chi ,
\end{aligned}\;}\tag{4.24}
\end{equation*}
read row-by-row off $\mathsf E^{-1}$ (4.12). One leftover term $\partial_s(\sin\theta\,\mathsf E^{-1})$ remains --- next.
\end{frame}

\begin{frame}{The divergence identity makes it clean}
\setstage{3}
Expanding the outer derivative gives the clean double-derivative \emph{plus} a first-order remainder:
\begin{equation*}
\hat T_{\mathrm{rot}}=\tfrac12\mu_{\alpha\beta}\hat J_\alpha\hat J_\beta
-\tfrac{\hbar^2}{2}\mu_{\alpha\beta}\underbrace{\Bigl[\tfrac{1}{\sin\theta}\partial_s\!\bigl(\sin\theta(\mathsf E^{-1})_{s\alpha}\bigr)\Bigr]}_{\text{divergence (4.26)}}(\mathsf E^{-1})_{t\beta}\partial_t .
\end{equation*}
The divergence vanishes term by term:
\begin{align*}
\alpha{=}1:&\ \partial_\phi(-\cos\chi)+\partial_\theta(\sin\theta\sin\chi)+\partial_\chi(\cos\theta\cos\chi)=0,\\
\alpha{=}2:&\ \partial_\phi(\sin\chi)+\partial_\theta(\sin\theta\cos\chi)+\partial_\chi(-\cos\theta\sin\chi)=0,\quad \alpha{=}3:\ \partial_\chi\sin\theta=0.
\end{align*}
\begin{alertblock}{Why the rotation block quantizes so cleanly (for the advanced reader)}
Podolsky's rule \emph{should} spawn $\partial\sqrt g$ and $\partial g^{\Omega\Omega}$ first-order terms --- but they collapse into this one divergence, zero because the rotation generators are divergence-free on $\sin\theta\,d\Omega$. Hence $\hat T_{\mathrm{rot}}=\tfrac12\mu_{\alpha\beta}\hat J_\alpha\hat J_\beta$, Hermitian, \alert{no pseudopotential} (4.29).
\end{alertblock}
\end{frame}

\begin{frame}{The Coriolis (rotation--vibration) block --- also clean}
\setstage{3}
The off-diagonal blocks $g^{\Omega Q}=-\mathsf E^{-1}\boldsymbol\mu\boldsymbol\sigma$ give the cross term (4.30):
\begin{equation*}
\hat T_{\mathrm{Cor}}=-\tfrac12\Bigl[\hat J_\alpha\,\mu_{\alpha\beta}\,\hat\pi_\beta
+\tfrac{1}{\sqrt\gamma}\,\hat P_b\,\sqrt\gamma\;\sigma_{\alpha b}\mu_{\alpha\beta}\,\hat J_\beta\Bigr],\qquad \hat\pi_\alpha=\textstyle\sum_b\sigma_{\alpha b}(Q)\hat P_b .
\end{equation*}
The two groups are \alert{adjoints of each other} on the measure (4.21) --- the $\gamma^{\pm1/2}$ are exactly what integration by parts over $\sqrt\gamma\,dQ$ produces --- so their sum is Hermitian \emph{automatically}, with no residue.
\begin{block}{Section result}
Rotation (4.29) and Coriolis (4.30) quantize with \textbf{no} pseudopotential --- the divergence identity and the adjoint-pair structure do all the work. The flat-measure rescaling of the next section leaves both untouched (giving the symmetric $-\tfrac12(\hat J\mu\hat\pi+\hat\pi\mu\hat J)$, 4.35). \alert{Every pseudopotential comes from the vibrational block.}
\end{block}
\end{frame}

\section{The vibrational block \& the Watson pseudopotential}

\begin{frame}{The vibrational block is different}
\setstage{3}
The vibrational inverse-metric block is $g^{QQ}=\mathbb 1+\boldsymbol\sigma^{\mathsf T}\boldsymbol\mu\boldsymbol\sigma$; the LB operator on the curved measure $\sqrt\gamma$ is
\begin{equation*}
\hat T_{\mathrm{vib}}=-\frac{\hbar^2}{2}\,\frac{1}{\sqrt\gamma}\,\partial_{Q_b}\!\bigl(\sqrt\gamma\,g^{QQ}_{bc}\,\partial_{Q_c}\bigr).
\end{equation*}
Expand the $\delta_{bc}$ part by one product rule:
\begin{equation*}
-\tfrac{\hbar^2}{2}\tfrac{1}{\sqrt\gamma}\partial_{Q_b}\!\bigl(\sqrt\gamma\,\partial_{Q_b}\psi\bigr)=\tfrac12\hat P_b^2\psi
\underbrace{-\tfrac{\hbar^2}{2}\bigl(\tfrac12\partial_{Q_b}\!\ln\gamma\bigr)\partial_{Q_b}\psi}_{\text{first-order remainder}} .
\end{equation*}
\alert{Unlike rotation, this remainder does not vanish}: $\partial_{Q_b}\ln\gamma\ne0$ because $\gamma=\det\mathbf I'(Q)$ genuinely depends on the shape, and the bare $\hat P_b$ is \emph{not} Hermitian on $\sqrt\gamma\,dQ$. How to handle it?
\end{frame}

\begin{frame}{Warm-up: the radial part of the spherical Laplacian}
\setstage{3}
Read the radial coordinate as an effective one-body \emph{vibration} (a diatomic bond). Its Laplace--Beltrami operator, on the curved measure $r^2\,dr$, is
\begin{equation*}
\hat T_r=-\frac{\hbar^2}{2m}\frac{1}{r^2}\partial_r(r^2\partial_r),\qquad \int|R|^2\,r^2\,dr=1 .
\end{equation*}
But vibrational wavefunctions are conventionally normalized on a \emph{flat} measure. Rescale $u=rR$ (i.e.\ $\psi\to g^{1/4}\psi$, $\sqrt g\propto r^2$) so $\int|u|^2\,dr=1$:
\begin{equation*}
\boxed{\;\hat T_r\ \xrightarrow{\ u=rR\ }\ -\frac{\hbar^2}{2m}\frac{d^2}{dr^2}\ \text{ on the flat measure } dr .\;}
\end{equation*}
The awkward $\tfrac{2}{r}\partial_r$ is gone --- absorbed into the rescaling. \emph{This} move is what, in general, leaves a pseudopotential behind.
\end{frame}

\begin{frame}{Two schemes: Laplace--Beltrami vs Podolsky}
\setstage{3}
The \emph{same} operator, in two representations:
\begin{center}\small
\begin{tabular}{@{}p{0.44\textwidth}|p{0.46\textwidth}@{}}
\textbf{Laplace--Beltrami} & \textbf{Podolsky (flat measure)}\\[2pt]
wavefn on curved $\sqrt g\,d^nq$ & $\psi_W=g^{1/4}\psi$ on flat $d^nq$\\[2pt]
$-\tfrac{\hbar^2}{2\sqrt g}\partial(\sqrt g\,g^{ij}\partial)$ & $\tfrac12\hat p_i g^{ij}\hat p_j+U$\\[2pt]
no explicit potential & extra $c$-number $U=$ \alert{pseudopotential}
\end{tabular}
\end{center}
Related by conjugation with $g^{1/4}$; $U$ is the \alert{bookkeeping cost of the flat measure}.
\begin{alertblock}{Radial case: $U=0$}
For $\gamma\propto r^2$ the residual cancels ($u=rR$ is exact) --- like the rigid rotor. A genuine shape-dependent $\gamma(Q)$ does \alert{not} cancel: that is where the Watson potential is born.
\end{alertblock}
\end{frame}

\begin{frame}{Rescaling the molecular vibrational block}
\setstage{3}
Same move with $\gamma=\det\mathbf I'(Q)$: absorb the shape factor, $\psi_W=\gamma^{1/4}\psi$, so the vibrational measure is flat, $\prod_b dQ_b$ (normal-mode space treated as flat). Conjugating $\hat T_{\mathrm{vib}}$ (one product rule per line):
\begin{align*}
\partial_{Q_b}(\gamma^{-1/4}\psi_W)&=\gamma^{-1/4}[\partial_{Q_b}\psi_W-\tfrac14(\partial_{Q_b}\!\ln\gamma)\psi_W],\\
\partial_{Q_b}\bigl(\gamma^{1/4}[\cdots]\bigr)&=\gamma^{1/4}[\partial_{Q_b}^2\psi_W-\tfrac1{16}(\partial_{Q_b}\!\ln\gamma)^2\psi_W-\tfrac14(\partial_{Q_b}^2\ln\gamma)\psi_W].
\end{align*}
The cross terms cancel; the bare $\tfrac12\sum_b\hat P_b^2$ survives \emph{plus} a multiplicative residue
\begin{equation*}
U_{\mathrm{vib}}=\frac{\hbar^2}{8}\sum_b\Bigl[\partial_{Q_b}^2\ln\gamma+\tfrac14(\partial_{Q_b}\!\ln\gamma)^2\Bigr]\quad(+\,U_{\mathrm{Cor}}\text{ from the }\boldsymbol\sigma^{\mathsf T}\boldsymbol\mu\boldsymbol\sigma\text{ dressing}).
\end{equation*}
\alert{Now the residue is nonzero} --- the promised pseudopotential.
\end{frame}

\begin{frame}{Collapse to the Watson trace}
\setstage{3}
The residue $U_{\mathrm{vib}}+U_{\mathrm{Cor}}$ collapses through two exact $Q$-calculus identities. \textbf{Jacobi:} $\partial_{Q_b}\!\ln\gamma=\operatorname{tr}(\boldsymbol\mu\,\partial_{Q_b}\mathbf I')$. \textbf{Watson's closed form (4.37):} $\mathbf I'=A(\mathbf I^{e})^{-1}A$ with $A=\mathbf I^{e}+\tfrac12\sum_b a^{b}Q_b$, so every derivative reduces to constants; the Eckart sum rules fix the remainder:
\begin{equation*}
\boxed{\;\hat U(Q)=-\frac{\hbar^2}{8}\sum_{\alpha=1}^{3}\mu_{\alpha\alpha}(Q).\;}\tag{4.38}
\end{equation*}
\begin{itemize}
\item\footnotesize A pure $c$-number from the \emph{measure} --- kernel ordering never makes one.
\item\footnotesize Electrons don't modify it ($\gamma,\boldsymbol\mu$ depend on $Q$ alone).
\item\footnotesize Special to \emph{rectilinear} normal coordinates (Watson's closed form).
\end{itemize}
\end{frame}

\section{The complete quantum Hamiltonian}

\begin{frame}{Electronic kinetic operator}
\setstage{4}
In BF Cartesian electron coordinates $\hat{\bar{\mathbf p}}_i=-i\hbar\nabla_{\bar{\mathbf r}_i}$ is already Hermitian on the flat measure, so the electron + mass-polarization piece quantizes directly:
\begin{equation*}
\hat T_e=\sum_i\frac{\hat{\bar{\mathbf p}}_i^{\,2}}{2m_e}+\frac{1}{2M_N}\Bigl(\sum_i\hat{\bar{\mathbf p}}_i\Bigr)^{2}
=\sum_i\frac{\hat{\bar{\mathbf p}}_i^{\,2}}{2\mu_e}+\frac{1}{M_N}\sum_{i<l}\hat{\bar{\mathbf p}}_i\!\cdot\!\hat{\bar{\mathbf p}}_l. \tag{4.39, 4.40}
\end{equation*}
The diagonal part renormalizes the electron mass, $\tfrac1{m_e}+\tfrac1{M_N}=\tfrac1{\mu_e}$; the off-diagonal two-electron term is the mass polarization \emph{sensu stricto} (nuclear recoil).
\end{frame}

\begin{frame}{The exact non-relativistic molecular Hamiltonian}
\setstage{4}
Collecting the flat-measure blocks --- rotational (4.29), Coriolis (4.35), vibrational, pseudopotential (4.38), electronic (4.39), and the shift $\hat J_\alpha\to\hat J_\alpha-\hat L_{\mathrm{elec},\alpha}$:
\begin{equation*}
\boxed{
\begin{aligned}
\hat H_{\mathrm{int}}=\;&\tfrac12\!\sum_{\alpha\beta}\bigl(\hat J_\alpha-\hat L_{\mathrm{elec},\alpha}-\hat\pi_\alpha\bigr)\mu_{\alpha\beta}(Q)\bigl(\hat J_\beta-\hat L_{\mathrm{elec},\beta}-\hat\pi_\beta\bigr) &&\text{\footnotesize rotation--vibration--electron}\\[-1pt]
&+\tfrac12\!\sum_b\hat P_b^2 &&\text{\footnotesize vibration}\\[-1pt]
&+\sum_i\frac{\hat{\bar{\mathbf p}}_i^{\,2}}{2m_e}+\frac{1}{2M_N}\Bigl(\sum_i\hat{\bar{\mathbf p}}_i\Bigr)^{2} &&\text{\footnotesize electrons + mass pol.}\\[-1pt]
&-\frac{\hbar^2}{8}\sum_\alpha\mu_{\alpha\alpha}(Q) &&\text{\footnotesize Watson pseudopotential}\\[-1pt]
&+V_{NN}+V_{Ne}+V_{ee} &&\text{\footnotesize Coulomb}
\end{aligned}}\tag{4.41}
\end{equation*}
\centering\footnotesize on $L^2(d\mu_W)$, $d\mu_W=\sin\theta\,d\phi d\theta d\chi\prod_b dQ_b\prod_i d^3\bar{\mathbf r}_i$. Every operator is an explicit differential operator, Hermitian by integration by parts.
\end{frame}

\begin{frame}{The journey, in four stages}
\setstage{4}
\begin{enumerate}
  \item[\textbf{1.}] \alert{Classical}: lab $H$ (1.1) $\to$ one mixed mass-centered point transformation $\to$ $H$ (2.12): COM exactly split for any $\mathbf P_{\mathrm{cm}}$, mass polarization included --- no boost, no rest frame.
  \item[\textbf{2.}] \alert{Body-Fixed}: velocity-form Lagrangian ($\mathbf I_N+\mathbf I_e-\mathbf I_X$) $\to$ key identity $\mathbf J-\mathbf L_{\mathrm{elec}}=\mathbf I_N\boldsymbol\omega+\mathbf L_N^{\mathrm{vib}}$ ($\mathbf I_e,\mathbf I_X$ cancel) $\to$ classical $H_{\mathrm{int}}$ (3.22) with Watson $\boldsymbol\mu=(\mathbf I_N-\boldsymbol\sigma\boldsymbol\sigma^{\mathsf T})^{-1}$.
  \item[\textbf{3.}] \alert{Quantize}, coordinate-natively: metric (4.15) $\to$ measure (4.21) $\to$ inverse (4.22) $\to$ Podolsky blocks $\to$ $\gamma^{1/4}$ rescaling $\to$ Watson $\hat U=-\tfrac{\hbar^2}{8}\sum_\alpha\mu_{\alpha\alpha}$ (4.38). Hermiticity by integration by parts $+$ divergence identity (4.26) --- \alert{no angular-momentum algebra}.
  \item[\textbf{4.}] \alert{Quantum $\hat H$}: the exact non-relativistic $\hat H_{\mathrm{int}}$ (4.41).
\end{enumerate}

\begin{frame}{Lagrangians \& Hamiltonians: stages 1--2}
\setstage{4}
\small
\textbf{1. Lab-frame Cartesian.}
\begin{equation*}
\mathcal L=\sum_j\tfrac12 M_j|\dot{\mathbf R}_j|^2+\sum_i\tfrac12 m_e|\dot{\mathbf r}_i|^2-V,
\qquad
H=\sum_j\frac{\mathbf P_j^2}{2M_j}+\sum_i\frac{\mathbf p_i^2}{2m_e}+V .
\end{equation*}
\textbf{2. Mass-centered} (total CoM $\mathbf R_{\mathrm{cm}}$; internals referred to the nuclear CoM):
\begin{align*}
\mathcal L&=\tfrac12 M_{\mathrm{tot}}|\dot{\mathbf R}_{\mathrm{cm}}|^2+\sum_j\tfrac12 M_j|\dot{\mathbf R}'_j|^2+\sum_i\tfrac12 m_e|\dot{\mathbf r}'_i|^2-\frac{m_e^2}{2M_{\mathrm{tot}}}\Bigl(\sum_i\dot{\mathbf r}'_i\Bigr)^2-V,\\
H&=\frac{\mathbf P_{\mathrm{cm}}^2}{2M_{\mathrm{tot}}}+\sum_j\frac{\mathbf P_j^{\prime\,2}}{2M_j}+\sum_i\frac{\mathbf p_i^{\prime\,2}}{2m_e}+\frac{1}{2M_N}\Bigl(\sum_i\mathbf p'_i\Bigr)^2+V .
\end{align*}
\centering\footnotesize Mass polarization: velocity form $-\tfrac{m_e^2}{2M_{\mathrm{tot}}}(\sum\dot{\mathbf r}')^2$ $\ \leftrightarrow\ $ momentum form $+\tfrac{1}{2M_N}(\sum\mathbf p')^2$.
\end{frame}

\begin{frame}{Lagrangians \& Hamiltonians: stages 3--4}
\setstage{4}
\footnotesize
\textbf{3. Body-fixed (Eckart) Cartesian} --- electrons in velocity form:

&+\tfrac12\boldsymbol\omega^{T}(\mathbf I_N{+}\mathbf I_e{-}\mathbf I_X)\boldsymbol\omega

\end{align*}

\end{equation*}
\centering\footnotesize CoM decouples throughout ($\mathbf P_{\mathrm{cm}}^2/2M_{\mathrm{tot}}$); quantizing stage~4 gives $\hat H_{\mathrm{int}}$ (4.41).
