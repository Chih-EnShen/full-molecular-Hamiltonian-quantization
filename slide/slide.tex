\section{Motivation}

\begin{frame}{Why is the molecular Hamiltonian non-trivial?}
\setstage{0}
The lab Hamiltonian couples \emph{all} particles through the Coulomb potential:
\[
H=\sum_j\frac{\mathbf P_j^2}{2M_j}+\sum_i\frac{\mathbf p_i^2}{2m_e}+V .
\]
To do molecular physics we want to separate
\[
\underbrace{\text{translation}}_{\text{COM}}\;\big|\;\underbrace{\text{rotation}}_{\text{Euler angles }\Omega}\;\big|\;\underbrace{\text{vibration}}_{\text{normal coords }Q}\;\big|\;\underbrace{\text{electrons}}_{\bar{\mathbf r}_i}.
\]
\pause
\begin{alertblock}{The catch}
Rotation + vibration are \emph{curvilinear} coordinates. Naively quantizing
$p\to-i\hbar\,\partial$ is \textbf{not} Hermitian there. Getting it right forces the
Podolsky construction.
\end{alertblock}
\end{frame}

\begin{frame}{What we will \emph{not} approximate}
\setstage{0}
We keep everything exact until the very end:
\begin{itemize}
  \item \textbf{No} Born--Oppenheimer / clamped nuclei.
  \item \textbf{No} neglect of \emph{mass polarization}.
  \item \textbf{No} harmonic assumption.
\end{itemize}
\pause
The \emph{only} ingredients are:
\begin{enumerate}
  \item non-relativistic dynamics;
  \item the Eckart frame (a \emph{convention}, not an approximation);
  \item parametrizing the nuclear shape by $3N_{\mathrm{nucl}}-6$ normal coordinates (exact).
\end{enumerate}
\vspace{0.5ex}
\centering\itshape The endpoint is the exact quantum $\hat H_{\mathrm{int}}$ — the rigorous starting point
\end{frame}

\setstage{1}
\section{Classical Cartesian Hamiltonian}

\begin{frame}{The starting point (lab frame)}
\setstage{1}
$N_{\mathrm{nucl}}$ nuclei ($M_j,Z_je,\mathbf R_j,\mathbf P_j$) and $N_{\mathrm{elec}}$ electrons ($m_e,-e,\mathbf r_i,\mathbf p_i$), all in the space-fixed frame; SI units:
\begin{equation*}
H = \underbrace{\sum_{j}\frac{\mathbf{P}_j^{\,2}}{2M_j}}_{T_N} + \underbrace{\sum_{i}\frac{\mathbf{p}_i^{\,2}}{2m_e}}_{T_e} + V, \tag{1.1}
\end{equation*}
\begin{equation*}
V = \frac{1}{4\pi\epsilon_0}\left(\frac12\sum_{i\ne j}\frac{Z_i Z_j e^2}{R_{ij}} - \sum_{i,j}\frac{Z_j e^2}{|\mathbf{R}_j-\mathbf{r}_i|} + \frac12\sum_{i\ne j}\frac{e^2}{r_{ij}}\right). \tag{1.2}
\end{equation*}
Canonical brackets $\{R_{j\alpha},P_{k\beta}\}=\delta_{jk}\delta_{\alpha\beta}$, $\{r_{i\alpha},p_{j\beta}\}=\delta_{ij}\delta_{\alpha\beta}$.
\vspace{0.5ex}

\centering\footnotesize Everything that follows is an exact rewriting of (1.1)--(1.2).
\end{frame}

\setstage{1}
\section{Mass-centered coordinates}

\begin{frame}{Two reference points}
\setstage{1}
The translational coordinate and the internal reference point \alert{need not be the same}:
\begin{align*}
\mathbf{R}_{\mathrm{ncm}} &= \frac{1}{M_N}\sum_j M_j\mathbf{R}_j,
&\mathbf{R}_{\mathrm{cm}} &= \frac{1}{M_{\mathrm{tot}}}\Bigl(\sum_j M_j\mathbf{R}_j+m_e\sum_i\mathbf{r}_i\Bigr), \tag{2.1, 2.2}\\
\mathbf{R}'_j &= \mathbf{R}_j-\mathbf{R}_{\mathrm{ncm}},
&\mathbf{r}'_i &= \mathbf{r}_i-\mathbf{R}_{\mathrm{ncm}},\qquad \textstyle\sum_j M_j\mathbf{R}'_j=0. \tag{2.3, 2.4}
\end{align*}
\begin{block}{The mixed choice}
\textbf{Translation $=$ total CoM} $\Rightarrow$ correct total mass, and (we will see) the COM--internal cross terms cancel identically.\\
\textbf{Internals referred to the nuclear CoM} $\Rightarrow$ Born--Oppenheimer-friendly coordinates, $V$ form-invariant, and the mass polarization appears \emph{by itself}.
\end{block}
\end{frame}

\begin{frame}{Canonical momenta: mass-fraction share $+$ nuclear recoil}
\setstage{1}
The canonical momenta are written in terms of the internal momenta $\mathbf P'_j,\mathbf p'_i$:
\begin{equation*}
\boxed{\;\mathbf{P}_j = \frac{M_j}{M_{\mathrm{tot}}}\,\mathbf{P}_{\mathrm{cm}}+\mathbf{P}'_j-\frac{M_j}{M_N}\sum_l\mathbf{p}'_l,
\qquad \mathbf{p}_i = \frac{m_e}{M_{\mathrm{tot}}}\,\mathbf{P}_{\mathrm{cm}}+\mathbf{p}'_i.\;}\tag{2.8}
\end{equation*}
\begin{itemize}
  \item Each particle carries its \emph{total-mass-fraction} share of $\mathbf P_{\mathrm{cm}}$ plus its internal momentum.
  \item Each \emph{nucleus} additionally recoils against the electrons, $-\frac{M_j}{M_N}\sum_l\mathbf p'_l$, because the internal reference point (2.1) is built from the nuclei alone.
  \item Summing: mass fractions add to one, recoils cancel, $\sum_j\mathbf P_j+\sum_i\mathbf p_i=\mathbf P_{\mathrm{cm}}$. $\checkmark$
\end{itemize}
\end{frame}

\begin{frame}{The Hamiltonian lands directly on the final form}
\setstage{1}
Insert (2.8): every nucleus has $\mathbf P_j=M_j\mathbf b+\mathbf P'_j$ with the common vector $\mathbf b=\mathbf P_{\mathrm{cm}}/M_{\mathrm{tot}}-\sum_l\mathbf p'_l/M_N$ (2.9).
\\
Expanding, the $\mathbf P_{\mathrm{cm}}^2$ fractions sum to $\tfrac{1}{2M_{\mathrm{tot}}}$, the cross terms cancel
\alert{identically}, and squaring the recoil leaves the mass polarization:
\begin{equation*}
\boxed{\;H = \frac{\mathbf{P}_{\mathrm{cm}}^{\,2}}{2M_{\mathrm{tot}}}
+\underbrace{\sum_j\frac{\mathbf{P}_j^{'2}}{2M_j}+\sum_i\frac{\mathbf{p}_i^{'2}}{2m_e}
+\frac{1}{2M_N}\Bigl(\sum_i\mathbf{p}'_i\Bigr)^{\!2}+V(\mathbf R'_j,\mathbf r'_i)}_{\textstyle H_{\mathrm{int}}\ (2.13)}.\;} \tag{2.12}
\end{equation*}
\begin{itemize}
  \item $V$ unchanged in form: both internal families share the reference point $\mathbf R_{\mathrm{ncm}}$.
\end{itemize}
\centering\itshape $H_{\mathrm{int}}$ is what we carry into the body-fixed frame.
\end{frame}

\setstage{2}
\section{Body-fixed frame and the Routhian}

\begin{frame}{Body-fixed frame: rotate with the nuclei}
\setstage{2}
Introduce a rotation $\mathbf S(\Omega)$ (Euler angles $\Omega=\phi,\theta,\chi$) and body-fixed (BF) coordinates:
\begin{equation*}
\mathbf{R}'_j=\mathbf{S}(\Omega)\,\bar{\mathbf{R}}_j(Q),\qquad \mathbf{r}'_i=\mathbf{S}(\Omega)\,\bar{\mathbf{r}}_i. \tag{3.1}
\end{equation*}
The \emph{Eckart} conditions fix $\mathbf S$ from the \emph{nuclei}; the nuclear shape is rectilinear mass-weighted \alert{normal coordinates} $Q_b$:
\begin{equation*}
\bar{\mathbf{R}}_j(Q)=\bar{\mathbf{R}}_j^{\mathrm{eq}}+\frac{1}{\sqrt{M_j}}\sum_b\mathbf{l}_{jb}\,Q_b ,
\qquad \sum_j\mathbf{l}_{ja}\!\cdot\!\mathbf{l}_{jb}=\delta_{ab}. \tag{3.1a}
\end{equation*}
The BF angular velocity $\boldsymbol{\omega}$ is defined by
\begin{equation*}
(\mathbf{S}^{T}\dot{\mathbf{S}})_{\alpha\beta}=-\epsilon_{\alpha\beta\gamma}\,\omega_\gamma . \tag{3.2}
\end{equation*}
\end{frame}

\begin{frame}{A Routhian that keeps electrons in momentum form}
\setstage{2}
Start from $H$ (2.12). We need an action principle for the body-fixed \emph{nuclear} coordinates --- but the electron kinetic energy + mass polarization
$H_e=\sum_i\mathbf p_i'^{\,2}/2m_e+\tfrac{1}{2M_N}(\sum_i\mathbf p'_i)^2$
is a \emph{rotational scalar}. So we Legendre-transform only the COM and nuclear momenta and \alert{keep the electrons in momentum form}, giving the \textbf{Routhian}
\begin{equation*}
\boxed{\;\mathcal{R}=\tfrac12 M_{\mathrm{tot}}|\dot{\mathbf{R}}_{\mathrm{cm}}|^2+\tfrac12\sum_j M_j|\dot{\mathbf{R}}_j'|^2-H_e(\mathbf{p}'_i)-V.\;}\tag{3.6}
\end{equation*}
\begin{block}{Why a Routhian?}
$\mathcal R$ is a Lagrangian in the nuclear/COM coordinates $(\mathbf R_{\mathrm{cm}},\mathbf R'_j)$ and \emph{minus} a Hamiltonian in the electrons $(\mathbf r'_i,\mathbf p'_i)$: the electrons ride along as the scalar $H_e$, never converted to velocity form and back.
\end{block}
\end{frame}

\begin{frame}{Body-fixed Routhian}
\setstage{2}
$H_e$ is built from $\mathbf p'_i\!\cdot\!\mathbf p'_l$; under the passive rotation $\bar{\mathbf p}_i\equiv\mathbf S^{T}\mathbf p'_i$ it is \emph{unchanged} in form. Substituting the BF velocities $\dot{\mathbf R}'_j=\mathbf S(\boldsymbol{\omega}\!\times\!\bar{\mathbf R}_j+\dot{\bar{\mathbf R}}_j)$,
\begin{equation*}
\mathcal{R}=\tfrac12 M_{\mathrm{tot}}|\dot{\mathbf{R}}_{\mathrm{cm}}|^2
+\tfrac12\!\sum_j M_j|\boldsymbol{\omega}\!\times\!\bar{\mathbf{R}}_j+\dot{\bar{\mathbf{R}}}_j|^2
-\sum_i\frac{\bar{\mathbf{p}}_i^{\,2}}{2m_e}-\frac{1}{2M_N}\Bigl(\sum_i\bar{\mathbf{p}}_i\Bigr)^{2}-V. \tag{3.7}
\end{equation*}
Only the \emph{nuclear} kinetic term carries $\boldsymbol{\omega}$; $V$ is now manifestly $\mathbf S$-independent.
\end{frame}

\begin{frame}{Expand the nuclear term}
\setstage{2}
The vector identity (for any $\mathbf a,\dot{\mathbf a},\boldsymbol\omega$)
\begin{equation*}
\tfrac{m}{2}|\boldsymbol{\omega}\!\times\!\mathbf{a}+\dot{\mathbf{a}}|^{2}
=\underbrace{\tfrac{m}{2}\boldsymbol{\omega}^{T}(a^{2}\mathbb{1}-\mathbf{a}\mathbf{a}^{T})\boldsymbol{\omega}}_{\text{rotational}}
+\underbrace{m\,\boldsymbol{\omega}\!\cdot\!(\mathbf{a}\!\times\!\dot{\mathbf{a}})}_{\text{Coriolis}}
+\underbrace{\tfrac{m}{2}|\dot{\mathbf{a}}|^{2}}_{\text{vibrational}} \tag{3.9c}
\end{equation*}
\pause
applied to the nuclei ($m\!\to\!M_j$, $\mathbf a\!\to\!\bar{\mathbf R}_j$) gives the assembled Routhian
\begin{equation*}
\boxed{\;\mathcal{R}=\tfrac12 M_{\mathrm{tot}}|\dot{\mathbf{R}}_{\mathrm{cm}}|^2
+\underbrace{\tfrac12\boldsymbol{\omega}^{T}\mathbf{I}_N\boldsymbol{\omega}+\boldsymbol{\omega}\!\cdot\!\mathbf{L}_N^{\mathrm{vib}}+T_{\mathrm{vib}}^{N}}_{\text{nuclear}}-H_e-V.\;} \tag{3.13}
\end{equation*}
\alert{The rotational kernel is the \emph{nuclear} inertia $\mathbf I_N$ alone} --- no electronic inertia, because the electrons were never expanded in $\boldsymbol\omega$.
\end{frame}

\begin{frame}{Canonical momenta and the total angular momentum}
\setstage{2}
\textbf{Nuclear} angular momentum (from $\partial\mathcal R/\partial\boldsymbol\omega$):
\begin{equation*}
\mathbf{J}_N=\mathbf{I}_N\boldsymbol{\omega}+\mathbf{L}_N^{\mathrm{vib}}. \tag{3.15}
\end{equation*}
The BF frame rotates nuclei \emph{and} electrons, so the \alert{total} angular momentum is the sum [$\mathbf L_{\mathrm{elec}}=\sum_i\bar{\mathbf r}_i\!\times\!\bar{\mathbf p}_i$]:
\begin{equation*}
\mathbf{J}=\mathbf{J}_N+\mathbf{L}_{\mathrm{elec}}=\mathbf{I}_N\boldsymbol{\omega}+\mathbf{L}_N^{\mathrm{vib}}+\mathbf{L}_{\mathrm{elec}}. \tag{3.16}
\end{equation*}
Vibrational momentum, with the \alert{Coriolis vectors} $\boldsymbol\sigma_b$:
\begin{equation*}
P_b=\boldsymbol{\omega}\!\cdot\!\boldsymbol{\sigma}_b+\dot Q_b,\qquad
\boldsymbol{\sigma}_b=\sum_a\boldsymbol{\zeta}_{ab}Q_a,\quad \boldsymbol{\zeta}_{ab}=\sum_j\mathbf{l}_{ja}\!\times\!\mathbf{l}_{jb}. \tag{3.17a,b}
\end{equation*}
\end{frame}

\begin{frame}{The key identity}
\setstage{2}
Subtract the electronic part from (3.16):
\begin{equation*}
\boxed{\;\mathbf{J}-\mathbf{L}_{\mathrm{elec}}=\mathbf{I}_N(Q)\,\boldsymbol{\omega}+\mathbf{L}_N^{\mathrm{vib}}.\;} \tag{3.19}
\end{equation*}
\begin{block}{Why this matters}
The right side is \emph{purely nuclear}: the rotational kernel is $\mathbf I_N$, \textbf{not} the full inertia. All electronic + mass-polarization content sits on the left, in the kinematic shift $\mathbf J\to\mathbf J-\mathbf L_{\mathrm{elec}}$.
\end{block}
\pause
\footnotesize\emph{Cross-check:} the velocity-form route instead generates the electronic inertia $\mathbf I_e,\mathbf I_X$ in \emph{both} $\mathbf J$ and $\mathbf L_{\mathrm{elec}}$ --- they cancel exactly, reproducing (3.19). The Routhian makes the cancellation automatic.
\end{frame}

\begin{frame}{Watson reciprocal inertia}
\setstage{2}
Invert (3.19) for $\boldsymbol\omega$: write $\mathbf L_N^{\mathrm{vib}}=\boldsymbol\pi-\boldsymbol\sigma\boldsymbol\sigma^{\mathsf T}\boldsymbol\omega$ with
\begin{equation*}
\boldsymbol{\pi}=\sum_b\boldsymbol{\sigma}_bP_b\ \ (\text{field-free vibrational ang.\ mom.}),\qquad
\boldsymbol{\sigma}\boldsymbol{\sigma}^{\mathsf T}=\sum_b\boldsymbol{\sigma}_b\boldsymbol{\sigma}_b^{\mathsf T}. \tag{3.20a}
\end{equation*}
Then $\mathbf J-\mathbf L_{\mathrm{elec}}=(\mathbf I_N-\boldsymbol\sigma\boldsymbol\sigma^{\mathsf T})\boldsymbol\omega+\boldsymbol\pi$, so
\begin{equation*}
\boxed{\;\boldsymbol{\omega}=\boldsymbol{\mu}\bigl(\mathbf{J}-\mathbf{L}_{\mathrm{elec}}-\boldsymbol{\pi}\bigr),\qquad
\boldsymbol{\mu}\equiv(\mathbf{I}_N-\boldsymbol{\sigma}\boldsymbol{\sigma}^{\mathsf T})^{-1}.\;}\tag{3.20}
\end{equation*}
$\boldsymbol\mu$ is the \alert{Watson reciprocal-inertia tensor} --- built from \emph{nuclear} quantities only, with no Wilson $G$-matrix.
\end{frame}

\begin{frame}{The exact classical molecular Hamiltonian}
\setstage{2}
Legendre back; the rovibrational kinetic energy becomes $\tfrac12(\mathbf J-\mathbf L_{\mathrm{elec}}-\boldsymbol\pi)^{\mathsf T}\boldsymbol\mu(\mathbf J-\mathbf L_{\mathrm{elec}}-\boldsymbol\pi)+\tfrac12\sum_bP_b^2$:
\begin{equation*}
\boxed{
\begin{aligned}
H_{\mathrm{int}}=\;&\tfrac12\bigl(\mathbf{J}-\mathbf{L}_{\mathrm{elec}}-\boldsymbol{\pi}\bigr)^{\mathsf T}\boldsymbol{\mu}(Q)\bigl(\mathbf{J}-\mathbf{L}_{\mathrm{elec}}-\boldsymbol{\pi}\bigr)+\tfrac12\!\sum_bP_b^2\\
&+\sum_i\frac{\bar{\mathbf p}_i^{\,2}}{2m_e}+\frac{1}{2M_N}\Bigl(\sum_i\bar{\mathbf p}_i\Bigr)^{2}+V_{NN}+V_{Ne}+V_{ee}.
\end{aligned}}\tag{3.22}
\end{equation*}
\centering Exact, classical, in BF + Eckart coordinates. \alert{Now we quantize.}
\end{frame}

\setstage{3}
\section{The quantization problem}

\begin{frame}{Why naive quantization fails in curvilinear coordinates}
\setstage{3}
Label the curvilinear coordinates and write the kinetic energy as a quadratic form:
\begin{equation*}
\{q^A\}=\{\phi,\theta,\chi,\;Q_1,\dots,Q_f\},\quad f=3N_{\mathrm{nucl}}-6,\qquad
T=\tfrac12\,g^{AB}(q)\,p_Ap_B. \tag{4.1, 4.3}
\end{equation*}
Three objects control everything: the metric $g_{AB}$, its determinant $g$, and its inverse $g^{AB}$.
\begin{alertblock}{The problem, in one dimension}
On $L^2(w(q)\,dq)$ with a non-constant weight, integration by parts gives
\[
(-i\hbar\,\partial_q)^{\dagger}=-i\hbar\bigl(\partial_q+w'/w\bigr)\;\neq\;-i\hbar\,\partial_q .\tag{4.4}
\]
The curved measure $\sqrt{g}\,d^nq$ plays the role of $w$: naive $p_A\to-i\hbar\partial_A$ is \textbf{non-Hermitian}.
\end{alertblock}
\end{frame}

\begin{frame}{The Podolsky / Laplace--Beltrami prescription}
\setstage{3}
The kinetic operator Hermitian with respect to $\sqrt{g}\,d^nq$ \emph{by construction}:
\begin{equation*}
\boxed{\;\hat T=-\frac{\hbar^2}{2}\,\frac{1}{\sqrt{g}}\,\partial_A\!\bigl(\sqrt{g}\,g^{AB}\,\partial_B\bigr),\qquad g=\det g_{AB}.\;}\tag{4.5}
\end{equation*}
One integration by parts gives the manifestly symmetric (and positive) form
\begin{equation*}
\int\varphi^{*}\,\hat T\psi\,\sqrt{g}\,d^{n}q
=\frac{\hbar^{2}}{2}\int(\partial_A\varphi)^{*}\,g^{AB}\,(\partial_B\psi)\,\sqrt{g}\,d^{n}q . \tag{4.6}
\end{equation*}
For $g_{AB}=\delta_{AB}$ it reduces to $-\tfrac{\hbar^2}{2}\nabla^2$.
\begin{block}{The whole program of this stage}
Compute $g_{AB}$, $\det g$, $g^{AB}$ for the coordinates (4.1) --- \emph{explicitly} --- then expand (4.5) block by block. Hermiticity is automatic at every step.
\end{block}
\end{frame}

\section{The ro-vibrational geometry}

\begin{frame}{Euler-angle kinematics, explicitly}
\setstage{3}
$zyz$ convention $\mathbf S=R_z(\phi)R_y(\theta)R_z(\chi)$ (4.7). Differentiating $\mathbf S$ gives $\boldsymbol\omega=\mathsf E(\Omega)\dot{\boldsymbol\Omega}$ and, solving the linear system, $\dot{\boldsymbol\Omega}=\mathsf E^{-1}\boldsymbol\omega$:
\begin{equation*}
\mathsf E=\begin{pmatrix}-\sin\theta\cos\chi&\sin\chi&0\\ \ \ \sin\theta\sin\chi&\cos\chi&0\\ \ \ \cos\theta&0&1\end{pmatrix},\qquad
\mathsf E^{-1}=\begin{pmatrix}-\frac{\cos\chi}{\sin\theta}&\frac{\sin\chi}{\sin\theta}&0\\[3pt] \ \ \sin\chi&\cos\chi&0\\[3pt] \ \ \frac{\cos\theta\cos\chi}{\sin\theta}&-\frac{\cos\theta\sin\chi}{\sin\theta}&1\end{pmatrix} \tag{4.10, 4.12}
\end{equation*}
with
\begin{equation*}
\det\mathsf E=-\sin\theta . \tag{4.11}
\end{equation*}
\footnotesize The $1/\sin\theta$ entries are the Euler coordinate singularity at the poles; they are compensated by $|\det\mathsf E|=\sin\theta$ in the volume element. \alert{(4.10)--(4.12) are the only Euler-angle facts used from here on.}
\end{frame}

\begin{frame}{The ro-vibrational metric tensor}
\setstage{3}
The classical BF kinetic energy $2T_{\mathrm{rv}}=\boldsymbol\omega^{T}\mathbf I_N\boldsymbol\omega+2\boldsymbol\omega^{T}\!\sum_b\boldsymbol\sigma_b\dot Q_b+\sum_b\dot Q_b^2$ (4.13), re-expressed in the \emph{coordinate} velocities via $\boldsymbol\omega=\mathsf E\dot{\boldsymbol\Omega}$, defines the metric:
\begin{equation*}
\boxed{\;g_{AB}=
\begin{pmatrix}
\mathsf E^{T}\mathbf I_N\,\mathsf E & \mathsf E^{T}\boldsymbol\sigma\\[3pt]
\boldsymbol\sigma^{T}\mathsf E & \mathbb 1_f
\end{pmatrix}\;}\tag{4.15}
\end{equation*}
\begin{itemize}
  \item Euler--Euler block: inertia pulled back through the kinematic matrix.
  \item Euler--mode block: Coriolis coupling $\boldsymbol\sigma_b=\sum_a\boldsymbol\zeta_{ab}Q_a$ (constant $\boldsymbol\zeta$'s).
  \item mode--mode block: \emph{identity} --- the gift of mass-weighted rectilinear normal coordinates.
\end{itemize}
Every entry is an explicit function of $(\theta,\chi,Q)$.
\end{frame}

\begin{frame}{Electrons: flat, constant, decoupled from the measure}
\setstage{3}
Including electron velocities, the full mass matrix has a \emph{constant} electron block $\mathsf D$. Eliminating it (Schur complement $=$ completing the square) removes all $\boldsymbol\omega$-dependence:
\begin{equation*}
\mathsf C\mathsf D^{-1}\mathsf C^{\mathsf T}=\mathbf I_e-\mathbf I_X
\;\Longrightarrow\;
\det\mathsf M=\underbrace{\det\mathsf D}_{\text{const}}\cdot\det\!\begin{pmatrix}\mathbf I_N&\boldsymbol\sigma\\ \boldsymbol\sigma^{\mathsf T}&\mathbb 1_f\end{pmatrix} \tag{4.17}
\end{equation*}
--- the same $\mathbf I_e/\mathbf I_X$ cancellation as the classical cross-check of \S\,3.6.
\begin{itemize}
  \item Electron measure is the flat $\prod_i d^3\bar{\mathbf r}_i$; $\hat{\bar{\mathbf p}}_i=-i\hbar\nabla_{\bar{\mathbf r}_i}$ is Hermitian on it.
  \item $\hat L_{\mathrm{elec},\alpha}$ is Hermitian \emph{without symmetrization}: its vector field is divergence-free,
  $\sum_{i\gamma}\partial_{\bar r_{i\gamma}}(\epsilon_{\alpha\beta\gamma}\bar r_{i\beta})=0$ (4.18).
\end{itemize}
\alert{The entire quantization problem lives in the ro-vibrational sector}; electrons enter only via the flat kinetic operator and the shift $\hat J_\alpha\to\hat J_\alpha-\hat L_{\mathrm{elec},\alpha}$.
\end{frame}

\begin{frame}{Determinant and volume element}
\setstage{3}
Block row reduction gives $\det\bigl(\begin{smallmatrix}\mathsf a&\mathsf b\\ \mathsf b^{T}&\mathbb 1\end{smallmatrix}\bigr)=\det(\mathsf a-\mathsf b\mathsf b^{T})$ (4.19). Applied to (4.15):
\begin{equation*}
\boxed{\;\det g_{AB}=\sin^2\!\theta\;\gamma(Q),\qquad
\gamma\equiv\det\mathbf I',\qquad
\mathbf I'\equiv\mathbf I_N-\boldsymbol\sigma\boldsymbol\sigma^{T}.\;}\tag{4.20}
\end{equation*}
\begin{block}{The Watson-modified inertia appears by itself}
The determinant is governed by $\mathbf I'=\mathbf I_N-\boldsymbol\sigma\boldsymbol\sigma^{T}$ --- \emph{not} the bare $\mathbf I_N$: the Coriolis coupling reduces the rotational determinant. It is the same $\mathbf I'$ whose inverse $\boldsymbol\mu$ is the classical rotational kernel (3.20).
\end{block}
\begin{equation*}
\sqrt{g}\,d^{\,3+f}q=\underbrace{\sin\theta\,d\phi\,d\theta\,d\chi}_{\text{Haar measure on }SO(3)}\cdot\underbrace{\sqrt{\gamma(Q)}\,\textstyle\prod_bdQ_b}_{\text{rot.--vib.\ Jacobian}}\cdot\underbrace{\textstyle\prod_id^3\bar{\mathbf r}_i}_{\text{flat}} \tag{4.21}
\end{equation*}
\end{frame}

\begin{frame}{The inverse metric}
\setstage{3}
Claim, then \emph{verify} by block multiplication using only $\boldsymbol\mu\mathbf I'=\mathbb 1$ [(4.23)]:
\begin{equation*}
\boxed{\;g^{AB}=
\begin{pmatrix}
\mathsf E^{-1}\boldsymbol\mu\,\mathsf E^{-T} & -\mathsf E^{-1}\boldsymbol\mu\,\boldsymbol\sigma\\[3pt]
-\boldsymbol\sigma^{T}\boldsymbol\mu\,\mathsf E^{-T} & \mathbb 1_f+\boldsymbol\sigma^{T}\boldsymbol\mu\,\boldsymbol\sigma
\end{pmatrix},
\qquad \boldsymbol\mu=(\mathbf I_N-\boldsymbol\sigma\boldsymbol\sigma^{T})^{-1}.\;}\tag{4.22}
\end{equation*}
\begin{block}{Two inversions, one computation}
The Watson reciprocal inertia $\boldsymbol\mu$ --- introduced classically by inverting the momentum--velocity relation (3.20) --- reappears as the geometric inverse of the metric.
\end{block}
\footnotesize e.g.\ top-left: $\mathsf E^{T}\mathbf I_N\mathsf E\cdot\mathsf E^{-1}\boldsymbol\mu\mathsf E^{-T}-\mathsf E^{T}\boldsymbol\sigma\cdot\boldsymbol\sigma^{T}\boldsymbol\mu\mathsf E^{-T}=\mathsf E^{T}(\mathbf I_N-\boldsymbol\sigma\boldsymbol\sigma^{T})\boldsymbol\mu\mathsf E^{-T}=\mathbb 1_3$. $\checkmark$
\end{frame}

\section{Coordinate-native quantization}

\begin{frame}{$\hat J_\alpha$: defined, not postulated}
\setstage{3}
The expansion of (4.5) keeps producing $(\mathsf E^{-1})_{s\alpha}\partial_{\Omega_s}$. Name it:
\begin{equation*}
\boxed{\;
\hat J_\alpha\equiv-i\hbar\sum_s(\mathsf E^{-1})_{s\alpha}\partial_{\Omega_s}:\quad
\begin{aligned}
\hat J_1&=-i\hbar\bigl[-\tfrac{\cos\chi}{\sin\theta}\partial_\phi+\sin\chi\,\partial_\theta+\tfrac{\cos\theta\cos\chi}{\sin\theta}\partial_\chi\bigr],\\
\hat J_2&=-i\hbar\bigl[\ \ \tfrac{\sin\chi}{\sin\theta}\partial_\phi+\cos\chi\,\partial_\theta-\tfrac{\cos\theta\sin\chi}{\sin\theta}\partial_\chi\bigr],\\
\hat J_3&=-i\hbar\,\partial_\chi .
\end{aligned}\;}\tag{4.24}
\end{equation*}
\begin{alertblock}{The logic}
(4.24) is a \emph{definition} --- shorthand for three explicit combinations of Euler-angle derivatives. We will \textbf{never} compute a commutator $[\hat J_\alpha,\hat J_\beta]$ or use any angular-momentum algebra.
\end{alertblock}
\footnotesize Classical correspondence: $\tfrac12 p_Ag^{AB}p_B=\tfrac12\mathbf J^{T}\boldsymbol\mu\mathbf J-\mathbf J^{T}\boldsymbol\mu\boldsymbol\pi+\tfrac12\boldsymbol\pi^{T}\boldsymbol\mu\boldsymbol\pi+\tfrac12\sum_bP_b^2$ --- exactly the nuclear kernel of (3.22). (4.25)
\end{frame}

\begin{frame}{The divergence identity: the workhorse of this section}
\setstage{3}
The weighted divergence of each column of $\mathsf E^{-1}$ vanishes --- computed term by term from (4.12):
\begin{align*}
\alpha=1:\ \ &\partial_\phi(-\cos\chi)+\partial_\theta(\sin\theta\sin\chi)+\partial_\chi(\cos\theta\cos\chi)=0+\cos\theta\sin\chi-\cos\theta\sin\chi=0,\\
\alpha=2:\ \ &\partial_\phi(\sin\chi)+\partial_\theta(\sin\theta\cos\chi)+\partial_\chi(-\cos\theta\sin\chi)=0+\cos\theta\cos\chi-\cos\theta\cos\chi=0,\\
\alpha=3:\ \ &\partial_\chi(\sin\theta)=0. \tag{4.26}
\end{align*}
Hence each $\hat J_\alpha$ is \textbf{Hermitian on the Haar measure} $\sin\theta\,d\Omega$ by one integration by parts --- \emph{no symmetrization needed} (4.27).
\begin{alertblock}{This is the replacement}
(4.26) is the coordinate-level fact that replaces every appeal to angular-momentum algebra: the vector fields generating body-frame rotations are divergence-free with respect to the Haar measure.
\end{alertblock}
\end{frame}

\begin{frame}{The Podolsky operator, block by block}
\setstage{3}
Insert (4.21)--(4.22) into (4.5) and split into rotational, Coriolis, vibrational blocks (4.28):
\begin{itemize}
  \item \textbf{Rotational:} $\mu,\sqrt\gamma$ pass through $\partial_\Omega$; the divergence identity (4.26) kills the extra term:
  \[
  \hat T_{\mathrm{rot}}=\tfrac12\sum_{\alpha\beta}\mu_{\alpha\beta}(Q)\,\hat J_\alpha\hat J_\beta
  \qquad\text{--- Hermitian as written, \alert{no extra potential}.}\tag{4.29}
  \]
  \footnotesize (``A rigid rotor has no pseudopotential'' --- verified by three derivative computations, not by an algebra.)\normalsize
  \item \textbf{Coriolis:} the two groups are adjoints of each other on the measure (4.21) $\Rightarrow$ Hermitian sum, automatically (4.30).
  \item \textbf{Vibrational:} $g^{QQ}=\mathbb 1+\boldsymbol\sigma^{T}\boldsymbol\mu\boldsymbol\sigma$; the curved factor $\sqrt\gamma$ makes the bare $\hat P_b=-i\hbar\partial_{Q_b}$ non-Hermitian --- first-order remainders $\propto\partial_{Q_b}\ln\gamma$ appear (4.31). \alert{Removing them is the job of the rescaling.}
\end{itemize}
\end{frame}

\begin{frame}{Rescaling to the flat measure, step by step}
\setstage{3}
Absorb the shape factor into the wavefunction: $\psi_W\equiv\gamma^{1/4}\psi$, so $|\psi|^2\sqrt g\,dq=|\psi_W|^2d\mu_W$ (4.32). Conjugate the vibrational block by $\gamma^{1/4}$ (one product rule per line):
\begin{align*}
\partial_{Q_b}(\gamma^{-1/4}\psi_W)&=\gamma^{-1/4}\bigl[\partial_{Q_b}\psi_W-\tfrac14(\partial_{Q_b}\!\ln\gamma)\psi_W\bigr],\\
\gamma^{1/2}\partial_{Q_b}(\gamma^{-1/4}\psi_W)&=\gamma^{1/4}\bigl[\partial_{Q_b}\psi_W-\tfrac14(\partial_{Q_b}\!\ln\gamma)\psi_W\bigr],\\
\partial_{Q_b}\bigl(\gamma^{1/4}[\cdots]\bigr)&=\gamma^{1/4}\bigl[\partial_{Q_b}^2\psi_W-\tfrac{1}{16}(\partial_{Q_b}\!\ln\gamma)^2\psi_W-\tfrac14(\partial_{Q_b}^2\ln\gamma)\psi_W\bigr]. \tag{4.33}
\end{align*}
The cross terms cancel; what survives is the bare $\tfrac12\sum_b\hat P_b^2$ plus a \emph{multiplicative} residue $U_{\mathrm{vib}}$ (4.34).
\vspace{0.5ex}

The Coriolis block: the induced $\pm\tfrac{i\hbar}{4}(\partial_{Q_b}\ln\gamma)$ terms are equal and opposite $\Rightarrow$ they cancel, leaving the symmetric $-\tfrac12(\hat J_\alpha\mu_{\alpha\beta}\hat\pi_\beta+\hat\pi_\alpha\mu_{\alpha\beta}\hat J_\beta)$ with \alert{no residue} (4.35).
\end{frame}

\begin{frame}{Collapse to the Watson trace}
\setstage{3}
The total residue $U_{\mathrm{vib}}+U_{\mathrm{Cor}}$ collapses through two exact $Q$-calculus identities. \textbf{Jacobi:}
\begin{equation*}
\partial_{Q_b}\!\ln\gamma=\operatorname{tr}(\boldsymbol\mu\,\partial_{Q_b}\mathbf I'),\qquad
\partial_{Q_b}\boldsymbol\mu=-\boldsymbol\mu(\partial_{Q_b}\mathbf I')\boldsymbol\mu. \tag{4.36}
\end{equation*}
\textbf{Watson's closed form (4.37):} $\mathbf I'=A\,(\mathbf I^{e})^{-1}A$ with $A=\mathbf I^{e}+\tfrac12\sum_b a^{b}Q_b$ and \emph{constant} inertia derivatives $a^{b}$, so every derivative of $\gamma,\boldsymbol\mu$ reduces to constants. The gradient terms cancel; the Eckart sum rules fix the remainder:
\begin{equation*}
\boxed{\;\hat U(Q)=-\frac{\hbar^2}{8}\sum_{\alpha=1}^{3}\mu_{\alpha\alpha}(Q).\;}\tag{4.38}
\end{equation*}
\begin{itemize}
  \item\footnotesize A pure $c$-number from the \emph{measure} --- kernel ordering never generates one.
  \item\footnotesize The electrons do not modify it ($\gamma,\boldsymbol\mu$ depend on $Q$ alone; electronic Jacobian flat).
  \item\footnotesize The collapse is special to \emph{rectilinear} normal coordinates (the closed form (4.37)).
\end{itemize}
\end{frame}

\section{The complete quantum Hamiltonian}

\begin{frame}{Electronic kinetic operator}
\setstage{4}
In BF Cartesian electron coordinates $\hat{\bar{\mathbf p}}_i=-i\hbar\nabla_{\bar{\mathbf r}_i}$ is already Hermitian on the flat measure, so the electron + mass-polarization piece quantizes directly:
\begin{equation*}
\hat T_e=\sum_i\frac{\hat{\bar{\mathbf p}}_i^{\,2}}{2m_e}+\frac{1}{2M_N}\Bigl(\sum_i\hat{\bar{\mathbf p}}_i\Bigr)^{2}
=\sum_i\frac{\hat{\bar{\mathbf p}}_i^{\,2}}{2\mu_e}+\frac{1}{M_N}\sum_{i<l}\hat{\bar{\mathbf p}}_i\!\cdot\!\hat{\bar{\mathbf p}}_l. \tag{4.39, 4.40}
\end{equation*}
The diagonal part renormalizes the electron mass, $\tfrac1{m_e}+\tfrac1{M_N}=\tfrac1{\mu_e}$; the off-diagonal two-electron term is the mass polarization \emph{sensu stricto} (nuclear recoil).
\end{frame}

\begin{frame}{The exact non-relativistic molecular Hamiltonian}
\setstage{4}
Collecting the flat-measure blocks --- rotational (4.29), Coriolis (4.35), vibrational, pseudopotential (4.38), electronic (4.39), and the shift $\hat J_\alpha\to\hat J_\alpha-\hat L_{\mathrm{elec},\alpha}$:
\begin{equation*}
\boxed{
\begin{aligned}
\hat H_{\mathrm{int}}=\;&\tfrac12\!\sum_{\alpha\beta}\bigl(\hat J_\alpha-\hat L_{\mathrm{elec},\alpha}-\hat\pi_\alpha\bigr)\mu_{\alpha\beta}(Q)\bigl(\hat J_\beta-\hat L_{\mathrm{elec},\beta}-\hat\pi_\beta\bigr) &&\text{\footnotesize rotation--vibration--electron}\\[-1pt]
&+\tfrac12\!\sum_b\hat P_b^2 &&\text{\footnotesize vibration}\\[-1pt]
&+\sum_i\frac{\hat{\bar{\mathbf p}}_i^{\,2}}{2m_e}+\frac{1}{2M_N}\Bigl(\sum_i\hat{\bar{\mathbf p}}_i\Bigr)^{2} &&\text{\footnotesize electrons + mass pol.}\\[-1pt]
&-\frac{\hbar^2}{8}\sum_\alpha\mu_{\alpha\alpha}(Q) &&\text{\footnotesize Watson pseudopotential}\\[-1pt]
&+V_{NN}+V_{Ne}+V_{ee} &&\text{\footnotesize Coulomb}
\end{aligned}}\tag{4.41}
\end{equation*}
\centering\footnotesize on $L^2(d\mu_W)$, $d\mu_W=\sin\theta\,d\phi d\theta d\chi\prod_b dQ_b\prod_i d^3\bar{\mathbf r}_i$. Every operator is an explicit differential operator, Hermitian by integration by parts.
\end{frame}

\begin{frame}{Remark: the anomalous commutator --- not needed}
\setstage{4}
The traditional route to (4.41) (Klein 1929; Van Vleck 1951) derives the \emph{anomalous} algebra of the body-fixed angular momentum,
\begin{equation*}
[\hat J_\alpha,\hat J_\beta]=-\,i\hbar\,\epsilon_{\alpha\beta\gamma}\,\hat J_\gamma\qquad(\text{note the }-\text{ sign}),
\end{equation*}
and uses it to control the operator ordering of the rotational kernel.
\begin{block}{Nothing here used it}
\begin{itemize}
  \item $\hat J_\alpha$ entered only as the \emph{shorthand} (4.24) for explicit Euler-angle derivatives.
  \item Hermiticity came from integration by parts $+$ the divergence identity (4.26).
  \item The pseudopotential came from the $Q$-derivatives of the volume element.
\end{itemize}
The anomalous algebra is a correct property of the operators (4.24) --- but it is a \alert{consequence} of the coordinate representation, \alert{not an input} to the quantization.
\end{block}
\end{frame}

\begin{frame}{Born--Huang expansion and the BO truncation}
\setstage{4}
At fixed $Q$, the clamped-nuclei electronic problem
$\hat H_{\mathrm{el}}(Q)|\phi_n(Q)\rangle=E_n^{\mathrm{el}}(Q)|\phi_n(Q)\rangle$ (4.42--4.43) provides a complete parametric basis. The \alert{exact} molecular wavefunction expands as
\begin{equation*}
|\Psi\rangle=\sum_n|\chi_n\rangle\otimes|\phi_n(Q)\rangle \tag{4.45}
\end{equation*}
--- not an ansatz: it follows from completeness at each $Q$. Projecting the Schr\"odinger equation gives coupled channels (4.46) with non-adiabatic couplings $\hat\Lambda_{nm}$ (4.47).
\begin{block}{Born--Oppenheimer $=$ diagonal truncation}
Set $\hat\Lambda_{nm}\to0$: single-surface nuclear dynamics (4.48) on the potential energy surface
\[
V_n^{\mathrm{BO}}(Q)=E_n^{\mathrm{el}}(Q)+V_{NN}(Q)+\hat U(Q). \tag{4.49}
\]
A controlled approximation \emph{of an exact operator} --- breaking down at conical intersections.
\end{block}
\end{frame}
